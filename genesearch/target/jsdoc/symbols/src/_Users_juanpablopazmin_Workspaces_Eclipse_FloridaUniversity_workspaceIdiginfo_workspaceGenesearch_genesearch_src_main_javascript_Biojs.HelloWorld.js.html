1 /** 
  2  * This is the description of the HelloWorld component. Here you can set any HTML text 
  3  * for putting on the generated documentation.
  4  * 
  5  * @class
  6  * @extends Biojs
  7  * 
  8  * @author <a href="mailto:johncar@gmail.com">John Gomez</a>
  9  * @version 1.0.0
 10  * 
 11  * @requires <a href='http://code.jquery.com/jquery-1.6.4.js'>jQuery Core 1.6.4</a>
 12  * @dependency <script language="JavaScript" type="text/javascript" src="../biojs/dependencies/jquery/jquery-1.6.4.min.js"></script>
 13  * 
 14  * @param {Object} options An object with the options for HelloWorld component.
 15  *    
 16  * @option {string} target 
 17  *    Identifier of the DIV tag where the component should be displayed.
 18  *    
 19  * @option {string} [fontFamily='"Andale mono", courier, monospace'] 
 20  *    Font list to be applied to the component content.
 21  *  
 22  * @option {string} [fontColor="white"] 
 23  *    HTML color code for the font.
 24  *    
 25  * @option {string} [backgroundColor="#7BBFE9"] 
 26  * 	  Background color for the entire div content.
 27  * 
 28  * @option {Object} [selectionFontColor="white"] 
 29  * 	  This color will be used to change the font color of selected text.
 30  * 
 31  * @option {Object} [ selectionBackgroundColor="yellow"] 
 32  * 	  This color will be used to change the background of selected text.
 33  *     
 34  * @example 
 35  * var instance = new Biojs.HelloWorld({
 36  * 		target : "YourOwnDivId",
 37  * 		selectionBackgroundColor : '#99FF00'
 38  * });	
 39  * 
 40  */
 41 Biojs.HelloWorld = Biojs.extend (
 42 /** @lends Biojs.HelloWorld# */
 43 {
 44   constructor: function (options) {
 45 	  // In JavaScript �this� always refers to the �owner� of the function we're executing (http://www.quirksmode.org/js/this.html)
 46 	  // Let's preserve the reference to 'this' through the variable self. In this way, we can invoke/execute 
 47 	  // our component instead of the object where 'this' is being invoked/executed.
 48 	  var self = this;
 49 	  
 50 	  // For practical use, create an object with the main DIV container 
 51 	  // to be used in all of the code of our component
 52 	  this._container = jQuery("#"+self.opt.target);
 53 	  
 54 	  // Apply options values
 55 	  this._container.css({
 56 		  'font-family': self.opt.fontFamily, // this is one example of the use of self instead of this
 57 		  'background-color': self.opt.backgroundColor,
 58 		  'color': self.opt.fontColor,
 59 		  'font-size': '36px',
 60 		  'text-align': 'center',
 61 		  'vertical-align':'middle',
 62 		  'display': 'table-cell',
 63 		  'width': '597px',
 64 		  'height': '300px'		  
 65 	  });
 66 	
 67 	  // Disable text selection and
 68 	  // Change the selection mouse pointer  
 69 	  // from text to hand.
 70 	  this._container.css({
 71 		  '-moz-user-select':'none',
 72 		  '-webkit-user-select':'none',
 73 		  'user-select':'none'
 74 	  });
 75 
 76 	  // Set the content
 77 	  text = 'Hello World!';
 78 
 79 	  for( i=0; i< text.length; i++ ) {
 80 		  this._container.append('<span>' + text[i] + '</span>');
 81 	  }	   
 82 
 83 	  // Internal method to initialize the event of select 'Hello'  
 84 	  this._addSelectionTrigger();
 85 	  
 86 	  // Internal method to set the onClick event 
 87 	  this._addSimpleClickTrigger();
 88   },
 89 
 90   /**
 91    *  Default values for the options
 92    *  @name Biojs.HelloWorld-opt
 93    */
 94   opt: {
 95      target: "YourOwnDivId",
 96      fontFamily: '"Andale mono", courier, monospace',
 97      fontColor: "white",
 98      backgroundColor: "#7BBFE9",
 99      selectionFontColor: "black",
100      selectionBackgroundColor: "yellow"
101   },
102   
103   /**
104    * Array containing the supported event names
105    * @name Biojs.HelloWorld-eventTypes
106    */
107   eventTypes : [
108 	/**
109 	 * @name Biojs.HelloWorld#onClick
110 	 * @event
111 	 * @param {function} actionPerformed A function which receives an {@link Biojs.Event} object as argument.
112 	 * @eventData {Object} source The component which did triggered the event.
113 	 * @eventData {string} type The name of the event.
114 	 * @eventData {int} selected Selected character.
115 	 * @example 
116 	 * instance.onClick(
117 	 *    function( objEvent ) {
118 	 *       alert("The character " + objEvent.selected + " was clicked.");
119 	 *    }
120 	 * ); 
121 	 * 
122 	 * */
123 	 "onClick",
124 	 
125 	/**
126 	 * @name Biojs.HelloWorld#onHelloSelected
127 	 * @event
128 	 * @param {function} actionPerformed A function which receives an {@link Biojs.Event} object as argument.
129 	 * @eventData {Object} source The component which did triggered the event.
130 	 * @eventData {string} type The name of the event.
131 	 * @eventData {int} textSelected Selected text, will be 'Hello' obviously.
132 	 * @example 
133 	 * instance.onHelloSelected(
134 	 *    function( objEvent ) {
135 	 *       alert("The word " + objEvent.textSelected + " was selected.");
136 	 *    }
137 	 * ); 
138 	 * 
139 	 * */
140      "onHelloSelected"      
141   ], 
142   
143   /**
144    * Change the font size. Do nothing it no value is provided.
145    * 
146    * @param {string} [size] The new font size in pixels.
147    *
148    * @example 
149    * instance.setSize("72px");
150    */
151   setSize: function(size) {
152 	  if ( size != undefined ){
153 		  jQuery("#"+this.opt.target).css('font-size', size);
154 	  }
155   },
156   
157   _addSelectionTrigger: function() {
158 
159 	  var self = this;
160 	  var isMouseDown = false;
161 	  
162 	  // Create the CSS class called selected to change both background and color 
163 
164 	  jQuery('<style> .selected { '+
165 		  	'background-color:' + self.opt.selectionBackgroundColor + ';' +
166 		  	'color:' + self.opt.selectionFontColor +'; }</style>'
167 	  		).appendTo('head');
168 	  
169 	  //
170 	  // Add the click event to each character in the content
171 	  // But remember, we must to figure out when 'Hello' is selected only
172 	  this._container.find('span')
173 	  	.mousedown(function() {
174 	  		
175 	  		// Turn on the flag 
176 			isMouseDown = true;
177 			
178 			// A new selection is starting
179 			// Reset all by removing the CSS "selected" class if already applied
180 			self._container.children('span').removeClass('selected')
181 			
182 			// Apply the class for this span/character
183 			// NOTE: "this" refers to the internal object 'span'
184 			// NOT to the component's instance
185 			jQuery(this).addClass('selected');
186 				
187 	  	}).mouseover(function() {
188 	  		// Check if the mouse is being dragged  
189 			if (isMouseDown) {
190 				jQuery(this).addClass('selected');
191 			}	
192 		})
193 		.mouseup(function() {
194 			
195 			/// Turn off the flag 
196 			isMouseDown = false;
197 			
198 			var textSelected = '';
199 			
200 			// Get the entire selected word
201 			self._container.children('span.selected')
202 				.each(function(){
203 					textSelected += jQuery(this).text();
204 				});
205 			
206 			// Since requirements, only "Hello" word should be selected 
207 			// to raise the event
208 			if (textSelected == 'Hello') {
209 				self.raiseEvent('onHelloSelected', {
210 					textSelected : textSelected
211 				})
212 			}
213 		});
214   },
215   
216   _addSimpleClickTrigger: function () {
217 	  
218 	  var self = this;
219 	  
220 	  // Add the click event to each character in the content
221 	  this._container.find('span')
222 	  	.click( function(e) {
223 	  		// A letter was clicked!
224 	  		// Let's discover which one was it
225 	  		// TIP: e.target contains the clicked DOM node
226 	  		var selected = jQuery(e.target).text();
227 	  		
228 	  		// Create an event object 
229 	  		var evtObject = { "selected": selected };
230 	  		
231 	  		// We're ready to raise the event onClick of our component
232 	  		self.raiseEvent('onClick', evtObject);
233 	  	});
234   }
235   
236 });
