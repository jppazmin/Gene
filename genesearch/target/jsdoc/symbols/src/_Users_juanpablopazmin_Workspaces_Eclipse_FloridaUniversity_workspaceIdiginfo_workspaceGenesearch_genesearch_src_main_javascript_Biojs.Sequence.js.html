314     _requestSequence: function ( accession ) {
315 		var self = this;
316 
317 		this._contentDiv.addClass('loading');
318     	
319     	Biojs.console.log("Requesting sequence for: " + accession );
320 
321 		jQuery.ajax({ 
322 			url: self.opt.sequenceUrl,
323 			dataType: "xml",
324 			data: { segment: accession },
325 			success: function ( xml  ) {
326 				try {
327 					
328 					var sequenceNode = jQuery(xml).find('SEQUENCE:first');
329 					self.setSequence( sequenceNode.text(), sequenceNode.attr("id"), sequenceNode.attr("label") );
330 					
331 				} catch (e) {
332 					Biojs.console.log("Error decoding response data: " + e.message );
333 					self.clearSequence("No sequence available", "../biojs/css/images/warning_icon.png");
334 				}
335 
336 			},
337 			error: function (jqXHR, textStatus, errorThrown) {
338 				Biojs.console.log("Error decoding response data: " + textStatus );
339 				self.clearSequence("Error requesting the sequence to the server " + this.url , "../biojs/css/images/warning_icon.png");
340 			}
341 		});

450 	/**
451     * Highlights a region using the font color defined in {Biojs.Sequence#highlightFontColor} by default is red.
452     *
453     * @example
454     * // highlight the characters within the position 100 to 150, included.
455     * mySequence.addHighlight( { "start": 100, "end": 150, "color": "white", "background": "red", "id": "aaa" } );
456     * 
457     * @param {Object} h The highlight defined as follows:
458     * 	
459     * 
460     * @return {int} representing the id of the highlight on the internal array. Returns -1 on failure  
461     */
462 	addHighlight : function ( h ) {
463 		var id = '-1';
464 		var color = "";
465 		var background = "";
466 		var highlight = {};
467 		
468 		if ( h instanceof Object && h.start <= h.end ) {
469 			
470 			color = ( "string" == typeof h.color )? h.color : this.opt.highlightFontColor;
471 			background = ( "string" == typeof h.background )? h.background : this.opt.highlightBackgroundColor;
472 			id = ( "string" == typeof h.id )? h.id : (new Number(this._highlightsCount++)).toString();
473 			
474 			highlight = { "start": h.start, "end": h.end, "color": color, "background": background, "id": id };
475 			
476 			this._highlights.push(highlight);
477 			this._applyHighlight(highlight);
478 			this._restoreSelection(h.start,h.end);
479 		} 
480 		
481 		return id;
482 	},
483 	/* 
484      * Function: Biojs.Sequence._applyHighlight
485      * Purpose:  Apply the specified color and background to a region between 'start' and 'end'.
486      * Returns:  -
487      * Inputs: highlight -> {Object} An object containing the fields start (int), end (int), 
488      * 						color (HTML color string) and background (HTML color string).
489      */
490 	_applyHighlight: function ( highlight ) {		
491 		var seq = this._contentDiv.find('.sequence');
492 		for ( var i = highlight.start - 1; i < highlight.end; i++ ){
493 			jQuery(seq[i])
494 				.css({ 
495 					"color": highlight.color,
496 					"background-color": highlight.background})
497 				.addClass("highlighted");
498 		}
499 	},
500 	/* 
501      * Function: Biojs.Sequence._applyHighlights
502      * Purpose:  Apply the specified highlights.
503      * Returns:  -
504      * Inputs: highlights -> {Object[]} An array containing the highlights to be applied.
505      */
506 	_applyHighlights: function ( highlights ) {
507 		for ( var i in highlights ) {
508 			this._applyHighlight(highlights[i]);
509 		}
510 	},
511 	/* 
512      * Function: Biojs.Sequence._restoreHighlights
513      * Purpose:  Repaint the highlights in the specified region.
514      * Returns:  -
515      * Inputs: start -> {int} Start of the region to be restored.
516      * 		   end -> {int} End of the region to be restored.
517      */
518 	_restoreHighlights: function ( start, end ) {
519 		var h = this._highlights;
520 		
521 		// paint the region using default blank settings
522 		this._applyHighlight({
523 			"start": start, 
524 			"end": end, 
525 			"color": this.opt.fontColor, 
526 			"background": this.opt.backgroundColor 
527 		});
528 		// restore highlights in that region
529 		for ( var i in h ) {
530 			// interval intersects with highlight i ?
531 			if ( !( h[i].start > end || h[i].end < start ) ) {
532 				a = ( h[i].start < start ) ? start : h[i].start;
533 				b = ( h[i].end > end ) ? end : h[i].end;
534 				this._applyHighlight({
535 					"start": a, 
536 					"end": b, 
537 					"color": h[i].color, 
538 					"background": h[i].background 
539 				});
540 			}
541 		}
542 	},
543 	/* 
544      * Function: Biojs.Sequence._restoreSelection
545      * Purpose:  Repaint the current selection in the specified region. 
546      * 			 It is used in the case of any highlight do overriding of the current selection. 
547      * Returns:  -
548      * Inputs: start -> {int} Start of the region to be restored.
549      * 		   end -> {int} End of the region to be restored.
550      */
551 	_restoreSelection: function ( start, end ) {
552 		var sel = this.opt.selection;
553 		// interval intersects with current selection ?
554 		// restore selection
555 		if ( !( start > sel.end || end < sel.start ) ) {
556 			a = ( start < sel.start ) ? sel.start : start;
557 			b = ( end > sel.end ) ? sel.end : end;
558 			this._applyHighlight({
559 				"start": a, 
560 				"end": b, 
561 				"color": this.opt.selectionFontColor, 
562 				"background": this.opt.selectionColor
563 			});
564 		}
565 	},

578 	/**
579     * Remove a highlight.
580     *
581     * @example
582     * // Clear the highlighted characters within the position 100 to 150, included.
583     * mySequence.removeHighlight("spin1");
584     * 
585     * @param {string} id The id of the highlight on the internal array. This value is returned by method highlight.
586     */
587 	removeHighlight : function (id) {	
588 		var h = this._highlights;
589 		for ( i in h ) {
590 			if ( h[i].id == id ) {
591 				start = h[i].start;
592 				end = h[i].end;
593 				h.splice(i,1);
594 				
595 				this._restoreHighlights(start,end);
596 				this._restoreSelection(start,end);
597 				
598 				break;
599 			}
600 		}
601 	},

942      * Function: Biojs.Sequence._getHTMLRowAnnot
943      * Purpose:  Build an annotation
944      * Returns:  HTML of the annotation

946      * 			 annotation -> {Object} 
947      *  		 settings -> {Object}
948      */
949     _getHTMLRowAnnot : function (currentPos, annotation, settings) {
950     	var styleBegin = 'border-left:1px solid; border-bottom:1px solid; border-color:';
951     	var styleOn = 'border-bottom:1px solid; border-color:';

953     	
954     	var row = [];

956     	var spaceBetweenChars = (settings.spaceBetweenChars)? ' ' : '';    	
957     	var defaultColor = annotation.color;
958     	var id = annotation.id;
959     	for ( var pos=currentPos; pos < end ; pos++ ) {
960 			// regions
961 			for ( var r in annotation.regions ) {
962 				region = annotation.regions[r];
963 				
964 				spaceAfter = '';
965 				spaceAfter += (pos % settings.numColsForSpace == 0 )? ' ' : '';
966 				spaceAfter += spaceBetweenChars;
967 				
968 				color = ((region.color)? region.color : defaultColor);
969 				data = 'class="annotation '+id+'" id="'+id+'" color="'+color+'" pos="'+pos+'"';
970 				
971 				if ( pos == region.start ) {
972 					row[pos] = '<span style="'+styleBegin+color+'" '+data+'> ';
973 					row[pos] += spaceAfter;
974 					row[pos] += '</span>';
975 				} else if ( pos == region.end ) {
976 					row[pos] = '<span style="'+styleEnd+color+' " '+data+'> ';
977 					//row[pos] += spaceAfter;
978 					row[pos] += '</span>';
979 				} else if ( pos > region.start && pos < region.end ) {
980 					row[pos] = '<span style="'+styleOn+color+'" '+data+'> ';
981 					row[pos] += spaceAfter;
982 					row[pos] += '</span>';
983 				} else if (!row[pos]) {
984 					row[pos] = ' ';
985 					row[pos] += spaceAfter;
986 				}
987 			}
988 		}
989 
990        	var str = row.join("");
991     	
992     	return ( str.indexOf("span") == -1 )? "" : str;

1052 	_drawSequence : function(a, opt) {
1053 		var str = '';
1054
