1 /**
  2  * This component visualize a div in a upper layer that can be used to select an area of other component. 
  3  * Although is not biological related component, it is generic enough to be used in other components, that 
  4  * require to select a region, see chromosome example. Just one selector can be used for a HTML component.
  5  * <p class="warning">WARNING: The CSS value for the position will be changed to relative.</p>
  6  * 
  7  * @class
  8  * @extends Biojs
  9  * 
 10  * @author <a href="gustavoadolfo.salazar@gmail.com">Gustavo A. Salazar</a>
 11  * @version 1.0.0
 12  * 
 13  * @requires <a href='http://code.jquery.com/query-1.7.2.min.js'>jQuery Core 1.7.2</a>
 14  * @dependency <script language="JavaScript" type="text/javascript" src="../biojs/dependencies/jquery/jquery-1.7.2.min.js"></script>
 15  * 
 16  * @requires <a href='http://jqueryui.com/download/jquery-ui-1.8.20.custom.zip'>jQuery UI 1.8.2</a>
 17  * @dependency <script src="../biojs/dependencies/jquery/jquery-ui-1.8.2.custom.min.js" type="text/javascript"></script>
 18  *
 19  * @requires <a href='http://jqueryui.com/download/jquery-ui-1.8.20.custom.zip'>jQuery UI CSS 1.8.2</a>
 20  * @dependency <link rel="stylesheet" href="../biojs/dependencies/jquery/jquery-ui-1.8.2.css" />
 21  * 
 22  * @requires <a href='http://www.west-wind.com/weblog/posts/2008/Sep/12/jQuery-CSS-Property-Monitoring-Plugin-updated'>jquery.watcher.js</a>
 23  * @dependency <script language="JavaScript" type="text/javascript" src="../biojs/dependencies/jquery.watcher.js"></script>
 24  * 
 25  * @requires <a href='http://www.ebi.ac.uk/~jgomez/biojs/biojs/css/biojs.selector.css'>Selector CSS</a>
 26  * @dependency <link rel="stylesheet" href="../biojs/css/biojs.selector.css" />
 27  * 
 28  * @param {Object} options An object with the options for the Chromosome component.
 29  * 
 30  * @option {string} target
 31  *    Identifier of the DIV tag where the component should be displayed.
 32  * 
 33  * @option {boolean} resize_left
 34  *    to indicate if the selector can be resizable from the left side. default value: true
 35  * 
 36  * @option {boolean} resize_right
 37  *    to indicate if the selector can be resizable from the right side. default value: true
 38  * 
 39  * @option {boolean} resize_top
 40  *    to indicate if the selector can be resizable from the top side. default value: true
 41  * 
 42  * @option {boolean} resize_bottom
 43  *    to indicate if the selector can be resizable from the bottom side. default value: true
 44  *    
 45  * @option {array} area
 46  *    an array with 4 elements(integers) to indicate the start position of the area selector. 
 47  *    The first two elements indicate the X and Y position of the top-left corner, 
 48  *    the last 2 elements indicate the bottom-right position.
 49  *    The position are relative to the target element.  
 50  * 
 51  * @example
 52  * 	var instance = new Biojs.AreaSelector({
 53  * 		target: "YourOwnDivId",
 54  * 		area: [10,10,40,40]
 55  * 	});
 56 
 57  * 
 58  */
 59 
 60 Biojs.AreaSelector = Biojs.extend (
 61 /** @lends Biojs.AreaSelector# */
 62 {
 63 	constructor: function (options) {
 64 		var self=this;
 65 		$("#"+self.opt.target).css("position","relative");
 66 		var innerCode="<div class='selector'>";
 67 		if (self.opt.resize_left) 	innerCode += "<div class='scaler left'></div>";
 68 		if (self.opt.resize_right) 	innerCode += "<div class='scaler right'></div>";
 69 		if (self.opt.resize_top)		innerCode += "<div class='scaler top'></div>";
 70 		if (self.opt.resize_bottom)	innerCode += "<div class='scaler bottom'></div>";
 71 		innerCode += "<div>";
 72 		$("#"+self.opt.target).append(innerCode);
 73 
 74 
 75 		$("#"+self.opt.target+" .selector .right").draggable({
 76 			axis: "x",
 77 			start :function(event) {
 78 				self.updateScalers();
 79 				$(this).parent().css('border-right-width',"0px");
 80 			},
 81 			stop :function(event) {
 82 				$("#"+self.opt.target+" .selector").width(self._removePx($(this).css('left'))+5);
 83 				$(this).parent().css('border-right-width',"1px");
 84 				self.updateScalers();
 85 				self.raiseEvent('onRegionChanged', {
 86 					region : self.getCoveredArea()
 87 				});
 88 			}
 89 		});
 90 		$("#"+self.opt.target+" .selector .left").draggable({
 91 			axis: "x",
 92 			start :function(event) {
 93 				self.updateScalers();
 94 				$(this).parent().css('border-left-width',"0px");
 95 			},
 96 			stop :function(event) {
 97 				var delta=$(this).position().left+5;
 98 				$("#"+self.opt.target+" .selector").width($("#"+self.opt.target+" .selector").width()-delta);
 99 				$("#"+self.opt.target+" .selector").css('left',($(this).parent().position().left+delta)+"px");
100 				$(this).parent().css('border-left-width',"1px");
101 				self.updateScalers();
102 				self.raiseEvent('onRegionChanged', {
103 					region : self.getCoveredArea()
104 				});
105 			}
106 		});
107 		$("#"+self.opt.target+" .selector .bottom").draggable({
108 			axis: "y",
109 			start :function(event) {
110 				self.updateScalers();
111 				$(this).parent().css('border-bottom-width',"0px");
112 			},
113 			stop :function(event) {
114 				$("#"+self.opt.target+" .selector").height(self._removePx($(this).css('top'))+5);
115 				$(this).parent().css('border-bottom-width',"1px");
116 				self.updateScalers();
117 				self.raiseEvent('onRegionChanged', {
118 					region : self.getCoveredArea()
119 				});
120 			}
121 		});
122 		$("#"+self.opt.target+" .selector .top").draggable({
123 			axis: "y",
124 			start :function(event) {
125 				self.updateScalers();
126 				$(this).parent().css('border-top-width',"0px");
127 			},
128 			stop :function(event) {
129 				var delta=$(this).position().top+5;
130 				$("#"+self.opt.target+" .selector").height($("#"+self.opt.target+" .selector").height()-delta);
131 				$("#"+self.opt.target+" .selector").css('top',($(this).parent().position().top+delta)+"px");
132 				$(this).parent().css('border-top-width',"1px");
133 				self.updateScalers();
134 				self.raiseEvent('onRegionChanged', {
135 					region : self.getCoveredArea()
136 				});
137 			}
138 		});
139 
140 		
141 		$("#"+self.opt.target).watch("left,top,width,height,display", function() {                         
142 			self.updateScalers();
143 		}, 100, "_containerMove");
144 		
145 		self.updateScalers();
146 	    self.setCoveredArea(self.opt.area);
147 	},
148 
149 	/**
150 	 * Default values for the options
151 	 * @name Biojs.AreaSelector-opt
152 	 */
153 	opt: {
154 		target: "YourOwnDivId",
155 		resize_left: true,
156 		resize_right: true,
157 		resize_top: true,
158 		resize_bottom: true,
159 		area: [0,0,50,50]
160 	},
161 
162 	/**
163 	 * Array containing the supported event names
164 	 * @name Biojs.AreaSelector-eventTypes
165 	 */	
166 	eventTypes: [
167 	    		 /**
168 	    		  * @name Biojs.AreaSelector#onRegionChanged
169 	    		  * @event
170 	    		  * @param {function} actionPerformed A function which receives an {@link Biojs.Event} object as argument.
171 	    		  * 	It gets activated when the covered area has changed
172 	    		  * @eventData {Object} source The component which did triggered the event.
173 	    		  * @eventData {string} type The name of the event.
174 	    		  * @eventData {array} region A 4-element array indicating the current position of the selector.
175 	    		  * The first two elements indicate the X and Y position of the top-left corner, 
176 	    		  * the last 2 elements indicate the bottom-right position.
177 	    		  * The position are relative to the target element.  
178 				  *    
179 	    		  * @example 
180 	    		  * 	instance.onRegionChanged(function( objEvent ) {
181 	    		  * 		alert("The current area is from the point [" + objEvent.region[0] + "," + objEvent.region[1] + "] to the point [" + objEvent.region[2] + "," + objEvent.region[3] + "] "); 
182 	    		  * 	}); 
183 	    		  */
184 	             "onRegionChanged"
185 	             ],
186 
187 	 /**
188 	  * get the current value of the area
189 	  * 
190 	  * @example 
191 	  * var area = instance.getCoveredArea();
192 	  * alert("The current area is from the point [" + area.region[0] + "," + area.region[1] + "] to the point [" + area.region[2] + "," + area.region[3] + "] ");	  
193 	  * 
194 	  */ 
195 	getCoveredArea: function(){
196 		var self=this;
197 		return [
198 		        	$("#"+self.opt.target+" .selector").position().left,
199 		        	$("#"+self.opt.target+" .selector").position().top,
200 		        	$("#"+self.opt.target+" .selector").position().left + $("#"+self.opt.target+" .selector").width(),
201 		        	$("#"+self.opt.target+" .selector").position().top + $("#"+self.opt.target+" .selector").height()
202 		        ];
203 	},
204 	 /**
205 	  * set the value of the area
206 	  * 
207 	  * @param {string} area  A 4-element array indicating the position to locate the selector.
208 	  * The first two elements indicate the X and Y position of the top-left corner, 
209 	  * the last 2 elements indicate the bottom-right position.
210 	  * The position are relative to the target element.  
211 	  * 
212 	  * @example 
213 	  * instance.setCoveredArea([ 20,20,50,50]);
214 	  * 
215 	  */ 
216 	setCoveredArea: function(area){
217 		var self=this;
218 		$("#"+self.opt.target+" .selector").css('left',area[0]+"px");
219 		$("#"+self.opt.target+" .selector").css('top',area[1]+"px");
220 		$("#"+self.opt.target+" .selector").width(area[2]-area[0]);
221 		$("#"+self.opt.target+" .selector").height(area[3]-area[1]);
222 		self.updateScalers();
223 		self.raiseEvent('onRegionChanged', {
224 			region : self.getCoveredArea()
225 		});
226 	},
227 	
228 	 /**
229 	  * @private
230 	  * Private: update the positions and the limits of the scalers
231 	  */
232 	updateScalers: function(){
233 		var self=this;
234 		$("#"+self.opt.target+" .selector .left").css("left","-5px");
235 		$("#"+self.opt.target+" .selector .left").css("top",($("#"+self.opt.target+" .selector").height()/2)-5);
236 		$("#"+self.opt.target+" .selector .right").css("left",$("#"+self.opt.target+" .selector").width()-5);
237 		$("#"+self.opt.target+" .selector .right").css("top",($("#"+self.opt.target+" .selector").height()/2)-5);
238 		$("#"+self.opt.target+" .selector .top").css("left",($("#"+self.opt.target+" .selector").width()/2)-5);
239 		$("#"+self.opt.target+" .selector .top").css("top","-5px");
240 		$("#"+self.opt.target+" .selector .bottom").css("left",($("#"+self.opt.target+" .selector").width()/2)-5);
241 		$("#"+self.opt.target+" .selector .bottom").css("top",$("#"+self.opt.target+" .selector").height()-5);
242 		$("#"+self.opt.target+" .selector .right").draggable("option","containment",[
243 			$("#"+self.opt.target+" .selector").offset().left-3,
244 			$("#"+self.opt.target).offset().top,
245 			$("#"+self.opt.target).offset().left+$("#"+self.opt.target).width()-4,
246 			$("#"+self.opt.target).offset().top+$("#"+self.opt.target).height()
247 		]);
248 		$("#"+self.opt.target+" .selector .left").draggable("option","containment",[
249  			$("#"+self.opt.target).offset().left-4,
250  			$("#"+self.opt.target).offset().top,
251 			$("#"+self.opt.target+" .selector").offset().left+$("#"+self.opt.target+" .selector").width()-5,
252  			$("#"+self.opt.target).offset().top+$("#"+self.opt.target).height()
253  		]);
254 		$("#"+self.opt.target+" .selector .bottom").draggable("option","containment",[
255  			$("#"+self.opt.target).offset().left,
256  			$("#"+self.opt.target+" .selector").offset().top-3,
257  			$("#"+self.opt.target).offset().left+$("#"+self.opt.target).width(),
258  			$("#"+self.opt.target).offset().top+$("#"+self.opt.target).height()-4
259  		]);
260  		$("#"+self.opt.target+" .selector .top").draggable("option","containment",[
261   			$("#"+self.opt.target).offset().left,
262   			$("#"+self.opt.target).offset().top-4,
263 			$("#"+self.opt.target).offset().left+$("#"+self.opt.target).width()-4,
264  			$("#"+self.opt.target+" .selector").offset().top+$("#"+self.opt.target+" .selector").height()-5
265   		]);
266 	},
267 	 /**
268 	  * @private
269 	  * Private: remove the last two characters of a string and return its integer value
270 	  */
271 	_removePx: function(str){
272 		return 1*str.substr(0,str.length-2);
273 	}
274 });
