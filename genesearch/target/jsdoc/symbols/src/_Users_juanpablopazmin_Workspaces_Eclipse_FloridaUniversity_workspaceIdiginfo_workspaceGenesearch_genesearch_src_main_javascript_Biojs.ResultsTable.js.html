213 	_requestUniprotAccession: function ( userQuery, dataSet, action ) {
214 		var self = this;
215 		var dataParams = [];
216 		var countStr;
217 		
218 		var strQueryAccession = 
219 			'SELECT distinct ?gene str(?uniprotAcc) as ?humanUniprotAcc '+
220 				'WHERE { '+
221 				   '?wikiArticle rdf:type gw_wiki:Category-3AHuman_proteins ; '+
222 				   'rdfs:label ?gene ; '+
223 				   'gw_property:Has_entrez_id ?entrezId ; '+
224 				   'gw_property:Has_uniprot_id ?uniprotAcc . '+
225 				   'FILTER (regex(?gene, "' + userQuery + '", "i")) . '+
226 			    '} LIMIT 1';
227 		
228 		//dataSet.url = dataSet.sparqlUrl;
229 		//dataSet.paramsMap = { "iDisplayStart": "firstResult", "iDisplayLength": "maxResults" };
230 		
231 		dataParams.push({ name: "url", value: dataSet.sparqlUrl });
232 		dataParams.push({ name: "query", value: escape(strQueryAccession) }); 
233 		dataParams.push({ name: "format", value: 'application/json' });
234 		
235 		Biojs.console.log("Requesting the accession: " + strQueryAccession);
236 		
237 		jQuery.ajax({ 
238 			url: dataSet.proxyUrl,
239 			dataType: "text",
240 			data: dataParams,
241 			success: function ( dataReceived ) {
242 				
243 				var accession = "";
244 				
245 				try {

249 				} catch ( e ) {
250 					Biojs.console.log("Error decoding response data: " + e.message );
251 				}
252 				
253 				dataSet.accessionId = accession;
254 				Biojs.console.log("Accession: " + accession );
255 				action.call( self, accession );

398     /**
399      * Rebuild the table
400      * 
401      * @example 
402      * myTable.setDataSource({
403      * 		sparqlUrl: '',
404      *  	proxyUrl: '../biojs/dependencies/proxy/proxy.php',
405      *  	query: "pubid:(10837477 OR 12029088)",
406      *      filter: false
407 	 * });
408      * 
409      */
410     setDataSource: function ( dataSet ) {
411      	   	
412 		// Remote data 
413     	if ( dataSet.hasOwnProperty("totalRecords") && dataSet.hasOwnProperty("url") ) {
414     		// Already have the number of records for pagination 
415     		// then, invoke parent's setDataSource
416     		this.base( dataSet );
417     		
418     	} else if ( this.opt.dataSet.query !== undefined ) {
419     		// Set the new query
420     		this.opt.dataSet = {};
421     		this.opt.dataSet.sparqlUrl = dataSet.sparqlUrl;
422     		this.opt.dataSet.proxyUrl = dataSet.proxyUrl;
423     		this.opt.dataSet.filter = dataSet.filter;
424     		this.opt.dataSet.query = escape(dataSet.query);
425     		this.opt.dataSet.queryWhere = dataSet.queryWhere;
426     		this.opt.dataSet.url = dataSet.sparqlUrl + '?query=' + this.opt.dataSet.query + "&format=application/json";
427     		this.opt.dataSet.paramsMap = { "iDisplayStart": "firstResult", "iDisplayLength": "maxResults" };
428     		// Request query counting rows and apply setDataSource then
429     		this._requestQueryCount( this.opt.dataSet, this.base );
430     	}

434 	 * Function: Biojs.InteractionTable._decodeToJSON
435 	 * Purpose:  Overrides the parent method to decode the received MITAB data into the expected JSON format. 
436 	 * Returns:  {Object} formatted in the expected JSON. 
437 	 * Inputs:   data -> {*} MITAB data received from the PSIQUIC server or another one. 
438 	 */
439 	_decodeToJSON: function ( dataReceived ) {
440 		Biojs.console.log("DATA DECODING FROM RDF RESULT to JSON");
441 
442 		var jsonData = { iTotalRecords: 0, iTotalDisplayRecords: 0, aaData: [] };
443 		var columns = Biojs.ResultsTable.COLUMNS;
444 		var decodedData, results, obj, articles;
445 			
446 		try {
447 			dataReceived = dataTable();
448 			//decodedData = jQuery.parseJSON( dataReceived );
449 			//results = decodedData.results.bindings;
450 			results = dataReceived.results.bindings;
451 
452 		} catch ( e ) {
453 			Biojs.console.log("Error decoding response data: " + e.message );
454 			return jsonData;
455 		}
456 		
457 		//Biojs.console.log(decodedData);
458 		Biojs.console.log(dataReceived);
459 		articles = {};
460 		
461 		// Group by title
462 		for ( r in results ) {
463 			obj = results[r];
464 			
465 			if ( ! articles.hasOwnProperty( obj.title.value ) ) {
466 				articles[ obj.title.value ] = new Array( columns.length );
467 				
468 				for ( c = 0; c < columns.length; c++ ) {
469 				    try {
470 				        articles[ obj.title.value ][ c ] = obj[ columns[c].key ].value;
471 
472 				    } catch (e) {
473 				        if ( c.optional ) {
474                             articles[ obj.title.value ][ c ] = "";
475 
476 				        } else {
477 				            articles[ obj.title.value ][ c ] = "";
478                             Biojs.console.log("Missing value for column: "+ c.name );
479 				        }
480 				    }
481 				}
482 				
483 			} else {
484 				// Concat multiple values
485 				for ( c = 0; c < columns.length; c++ ) {
486 					if ( ! columns[c].unique ) {
487 						if ( articles[ obj.title.value ][ c ].indexOf( obj[ columns[c].key ].value ) == -1 ) {
488 							articles[ obj.title.value ][ c ] += '; ' + obj[ columns[c].key ].value;
489 						}
490 					}
491 				}
492 			}
493 		}
494 
495 		// Add to data
496 		for ( title in articles ) {	
497 			jsonData.aaData.push( articles[ title ] );
498 		}
499 		
500 		jsonData.iTotalRecords = this.getTotalRecords();
501 		jsonData.iTotalDisplayRecords = this.getTotalRecords();
502 		
503 		Biojs.console.log(jsonData);
504 		
505 		this._jsonData = jsonData;
506 		
507 		return jsonData;
508 	},
