284 				Biojs.console.log("Error decoding response data: " + e.message );
285 				return jsonData;

294

297

311 					}
312 					
313 				} else {
314 					// Concat multiple values

438 	_paintLinksOf: function (resourceURI) {
439 		
440 		Biojs.console.log("Painting links of "+resourceURI);
441 		
442 		var nodeID = this._digest(resourceURI);
443 		var linkID, matchedItems;
444 		
445 		for ( var otherNodeID in this._resourceURI ) {
446 			
447 			// exclude itself
448 			if ( otherNodeID != nodeID ) {
449 				// create an identifier for the link   
450 	    		linkID = parseInt(nodeID,10) + parseInt(otherNodeID,10);
451 				
452 				// isn't the link already drawn?
453 	    		if ( this._links[linkID] == undefined ) {
454 	    			// Get matched items to create links 
455 	    			matchedItems = this._matchNodes(nodeID+'',otherNodeID+'');
456 	    			for ( var i in matchedItems ) {
457 	    				this.addLink( nodeID, otherNodeID, matchedItems[i] );
458 	    				// mark the link as created   
459 	    	    		this._links[linkID] = true;
460 	    			}
461 	    		}
462 
463 			} 
464     	}

586         			if ( (count++ >= minMatchedEntities) && (count >= this.opt.filterBy['frequency']) ) return true;

597     // djb2 algorithm (k=33) by Dan Bernstein
598     // returns a valid HTML id string from any string 
599     _digest: function( str ) {
600 	    var hash = 5381;
601 	    for (i = 0; i < str.length; i++) {
602 	        char = str.charCodeAt(i);
603 	        hash = ((hash << 5) + hash) + char; /* hash * 33 + c */
604 	    }
605 	    return hash.toString();  
606     }
