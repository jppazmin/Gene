1 /** 
  2  * PDBLogos shows a set of icons to give a quick summary of salient features of a PDB entry.
  3  * See more info on PDBLogos at http://pdbe.org/prints
  4  * 
  5  * @class
  6  * @extends Biojs
  7  * 
  8  * @author <a href="mailto:johncar@gmail.com">John Gomez-Carvajal</a>
  9  * @version 1.0.0
 10  *
 11  * @requires <a href='http://code.jquery.com/jquery-1.6.4.js'>Raphael 2.1.0</a>
 12  * @dependency <script language="JavaScript" type="text/javascript" src="../biojs/dependencies/graphics/raphael-2.1.0.js"></script>
 13  * 
 14  * @requires <a href='http://code.jquery.com/jquery-1.6.4.js'>jQuery Core 1.6.4</a>
 15  * @dependency <script language="JavaScript" type="text/javascript" src="../biojs/dependencies/jquery/jquery-1.4.2.min.js"></script>
 16  * 
 17  * @param {Object} options An object with the options for PDBLogos component.
 18  *    
 19  * @option {string} target
 20  *    Identifier/HTMLElement of the div container where the component should be rendered.
 21  *
 22  * @option {string} identifier
 23  *    PDB entry id for which prints will be rendered.
 24  *
 25  * @option {Object} raphael
 26  *    A Raphael object instead of div to render PDBprints into.
 27  *
 28  * @option {int} startX
 29  *    X-coordinate of point where rendering the array of PDBprints will start in the Raphael canvas
 30  *
 31  * @option {int} startY
 32  *    Y-coordinate of point point where rendering the array of PDBprints will start in the Raphael canvas
 33  *
 34  * @option {int} interval
 35  *    Gap size between successive PDBprints in a vertical or horizontal array
 36  *
 37  * @option {int} size
 38  *    Length of an icon logo
 39  * 
 40  * @option {string} [color="#588FE5"]
 41  *    Color for background as HTML color code.
 42  *
 43  * @option {char} orient
 44  *    Rendering orientation either vertical or horizontal
 45  * 
 46  * @option {string} [pdb3DIcon='biojs/css/images/3d.png']
 47  *    Image URL to be used as 3D structure's icon
 48  *    
 49  * @example 
 50  * var instance = new Biojs.PDBLogos({
 51  *    target: "YourOwnDivId",
 52  *    identifier: "1cbs"
 53  * });	
 54  * 
 55  */
 56 Biojs.PDBLogos = Biojs.extend (
 57 /** @lends Biojs.PDBLogos# */
 58 {
 59   /**
 60    *  Default values for the options
 61    *  @name Biojs.PDBLogos-opt
 62    */
 63   opt: {
 64     target: "YourOwnDivId",
 65 	raphael: null,
 66 	identifier:"",
 67 	startX: 10,
 68 	startY: 10,
 69 	interval: 10,
 70 	size: 40,
 71 	orient: 'h',
 72 	logosURL: 'http://www.ebi.ac.uk/pdbe-apps/widgets/pdbprints',
 73 	logos: [],
 74 	color: "#588FE5",
 75 	pdb3DIcon: 'biojs/css/images/3d.png'
 76   },
 77 
 78   /**
 79     * Array containing the supported event names
 80     * @name Biojs.PDBLogos-eventTypes
 81     */
 82   eventTypes : [
 83     /**
 84     * @name Biojs.PDBLogos#onClick
 85     * @event
 86     * @param {function} actionPerformed A function which receives an {@link Biojs.Event} object as argument.
 87     * @eventData {Object} source The component which did triggered the event.
 88     * @eventData {string} type The name of the event.
 89     * @eventData {string} identifier PDB actual entry.
 90     * @eventData {string} category Name of the PDBLogo category clicked.
 91     * @eventData {string} printsUrl Url of the PDB general information.
 92     * @example
 93     * instance.onClick(
 94     *    function( e ) {
 95     *       alert("Logo " + e.category + " of PDB " + e.identifier + " was clicked.");
 96     *    }
 97     * );
 98     *
 99     * */
100     "onClick"
101   ],
102 
103   constructor: function (options) {
104     this._initCanvas();
105     if( !Biojs.Utils.isEmpty(this.opt.identifier) ) {
106         this._doRequest(this.opt.identifier);
107     }
108   },
109 
110   _initCanvas: function() {
111     var config = this.opt;
112 
113     // Make Raphael object if not provided already
114     if ( !Biojs.Utils.isEmpty( config.raphael ) ){
115         config.raphael.remove();
116         config.raphael = null;
117     }
118 
119     var sizex = config.startX + config.size * Biojs.PDBLogos.CATEGORY.lenght;
120     var sizey = config.startY + (config.interval+config.size);
121 
122     if(config.orient == Biojs.PDBLogos.VERTICAL) {
123         sizey = config.startY + config.size * Biojs.PDBLogos.CATEGORY.lenght;
124         sizex = config.startX + (config.interval+config.size);
125     }
126     config.raphael = Raphael(config.target, sizex, sizey);
127   },
128 
129   /**
130     * Sets the pdb identifier generating a new request of logos.
131     *
132     * @param {string} value PDB entry.
133     *
134     * @example
135     * instance.setIdentifier("1j3s");
136     *
137   */
138   setIdentifier: function ( value ) {
139     this.opt.identifier = value;
140     this._initCanvas();
141     this._doRequest(value);
142   },
143 
144   _doRequest: function(identifier){
145      
146      var self = this;
147      
148      jQuery.ajax({
149         url: 'http://www.ebi.ac.uk/pdbe-apps/widgets/pdbprints',
150         data: { 'varname':'printsdata', 'pdbid': identifier, color: "transparent" },
151         dataType: 'script',
152         crossDomain: 'true',
153         type: 'GET',
154         success: function(response, callOptions) { self._printsLayout(printsdata); }
155      });
156   },
157 
158 
159   _printsLayout: function(data) {
160     var self = this;
161     var config = this.opt;
162     var ix = config.startX;
163     var iy = config.startY;
164     var i=0;
165     
166     config.logos = [];
167 	
168     data[config.identifier].PDBStructure = [ config.pdb3DIcon, 'structure'];
169 
170     for( category in Biojs.PDBLogos.CATEGORY ) {
171     	
172     	var isizeY = config.size, isizeX = config.size;
173         var imgUrl = data[config.identifier][category][0];
174         var printsURL = "http://www.pdbe.org";
175         
176         ix = config.startX;
177         iy = config.startY;
178         
179         if( config.orient == Biojs.PDBLogos.VERTICAL ) {
180             iy += config.size * (i++);
181         } else {
182         	ix += config.size * (i++);
183         }
184 
185         if(category == "PDBeLogo") {
186             config.logos.push(
187                 config.raphael.text(ix, iy + self.opt.size/4, config.identifier)
188                     .attr({
189                         'font-family':'Mono',
190                         'text-anchor':'start',
191                         'font-size': self.opt.size/3,
192                         'cursor':'pointer',
193                         'title':'View all about PDB ' + config.identifier
194                     })
195                     .data("identifier", config.identifier)
196                     .mouseup( function() {
197                         window.open( "http://www.pdbe.org/" + this.data("identifier") );
198                     })
199             );
200 
201             imgUrl = "http://www.ebi.ac.uk/pdbe-apps/widgets/html/PDBeWatermark_horizontal_128.png";
202             isizeX = config.size * 0.9;
203             isizeY = config.size/2;
204             iy += config.size/2;
205 
206         } else {
207         	
208         	printsURL += "/"+config.identifier+"/" + Biojs.PDBLogos.CATEGORY[category];
209 
210         	if ( category == "PDBStructure" ) {
211         		printsURL = "http://www.ebi.ac.uk/pdbe-srv/view/entry/" + config.identifier + "/openastex";
212         	}	
213             
214             // Add icon background
215             config.raphael.rect(ix, iy, isizeX, isizeY, 7).attr({'fill':self.opt.color});
216         }
217 
218         config.logos.push(
219             config.raphael.image(imgUrl, ix, iy, isizeX, isizeY)
220                 .attr({
221                 	'cursor':'pointer',
222                 	'title':'View '+category
223                 })
224                 .data({
225                     "printsURL": printsURL,
226                     "category": category
227                 })
228                 .mouseup( function() {
229                     self.raiseEvent( Biojs.PDBLogos.EVT_ON_CLICK, {
230                         printsURL: this.data("printsURL"),
231                         identifier: config.identifier,
232                         category: this.data("category")
233                     });
234                 })
235                 .mouseover( function() { } )
236         );
237 
238     }
239     
240   }
241 
242 },{
243     HORIZONTAL: 'h',
244     VERTICAL: 'v',
245 
246     CATEGORY: {
247        "PDBeLogo":        "",
248        "PrimaryCitation": "citation",
249        "Taxonomy":        "taxonomy",
250        "Expressed":       "taxonomy",
251        "Experimental":    "experimental",
252        "Protein":         "primary",
253        "NucleicAcid":     "primary",
254        "Ligands":         "ligands",
255        "PDBStructure":    "PDBStructure"
256     },
257 
258     EVT_ON_CLICK: "onClick"
259 
260 });
