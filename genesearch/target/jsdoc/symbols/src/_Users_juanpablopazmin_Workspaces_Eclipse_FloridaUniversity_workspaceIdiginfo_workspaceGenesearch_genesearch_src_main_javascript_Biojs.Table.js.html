518      _initSettings: function( settings ) {
519     	 var self = this;
520     	 
521     	 Biojs.console.log("initializing settings...");
522 
523     	 settings.aoColumnDefs = []; 
524 
525     	 this._setColumns(this.opt.columns, settings);
526     	 //this._setData(this.opt.dataSet, settings);
527 
528     	 // Hide columns 
529     	 if ( this.opt.hideColumns.length > 0 ) {
530     		 settings.aoColumnDefs.push( { "bVisible": false, "aTargets": this.opt.hideColumns } );
531     	 }
532 
533     	 // Order by 
534     	 if ( this.opt.orderBy.length > 0 ) {
535     		 settings.aaSorting = this.opt.orderBy;
536     	 }
537 
538     	 // Row selection column (checkboxes column) is enabled? 
539     	 if (this.opt.rowSelection) { 
540     		 var columnsToHide = this.opt.hideColumns;
541 
542     		 // Shift the index of the columns to hide
543     		 for ( i in columnsToHide ) {
544     			 columnsToHide[i] += 1;
545     		 }
546 
547     		 // Shift the index of the columns to order
548     		 for ( i in settings.aaSorting ) {
549     			 settings.aaSorting[i][0] += 1;
550     		 }
551     		 
552     		 // Disable the ordering on the column 0 (checkboxes column) 
553     		 settings.aoColumnDefs.push( { "bSortable": false, "aTargets": [ 0 ] } );
554     	 }
555     	 
556     	 // pagination
557     	 if ( !this.opt.paginate ) {
558     		 settings.bScrollInfinite = true;
559     		 settings.bScrollCollapse = true;
560 
561     	 } else {
562     		 settings.sPaginationType = "full_numbers"; 
563     		 settings.iDisplayLength = this.opt.pageLength;
564     		 settings.oLanguage = {
565     				 "oPaginate": {
566     					 "sPrevious": "<",
567     					 "sNext": ">",
568     					 "sFirst": "|",
569     					 "sLast": "|"
570     				 }
571     		 };
572     	 }
573     	 settings.bProcessing = true;
574     	 settings.bLengthChange = false;
575     	 //settings.sScrollX = this.opt.height;
576     	 settings.sScrollY = "100%";
577     	 
578     	 settings.fnRowCallback = function( nRow, aData, iDisplayIndex, iDisplayIndexFull ) { 
579     		 self.raiseEvent(Biojs.Table.EVT_ON_ROW_RENDERED, {
580     			 node: nRow,
581     			 data: aData,
582     			 index: iDisplayIndex
583     		 });
584     	 }
585 
586     	 settings.bAutoWidth = false;
587 
588

630      /* 
631       * Function: _fetchData
632       * Purpose:  do the Ajax request to get the data from server. Note that 'this' will not refered to the Biojs.Table instance
633       * 	due to this function will be linked to the internal DataTables object.
634       * Returns:  -
635       * Inputs:   
636       * 	sSource -> {string} HTTP source to obtain the data from (defined in option dataSet.url).
637       * 	aoData -> {Object[]} a key/value pair object containing the data to send to the server.
638       * 	fnCallback -> {function} to be called on completion of the data get process that will draw the data on the page.
639       * 	oSettings -> {object} DataTables settings object.
640       */
641      _fetchData: function ( sSource, aoData, fnCallback, oSettings ) {
642 
643     	 var httpRequest = { url: sSource };
644     	 var params = aoData;
645 
646     	 // Get the Biojs Table instance 
647     	 var biojsId = oSettings.sTableId.substr( "biojs_Table_".length );
648     	 var instance = Biojs.getInstance(biojsId);
649 
650     	 // Rename param names using those defined in the options instance.opt.dataSet.mapParams
651     	 instance._mapUrlParams(aoData);
652 
653     	 // Set callback function on success
654     	 //httpRequest.success = fnCallback;
655 
656     	 // Data type expected 
657     	 httpRequest.dataType = 'json';
658 
659     	 // Using proxy?
660     	 // Redirect using the proxy and encode all params as url data
661     	 if ( instance.getProxy() != undefined ) {
662 
663     		 // Redirect to proxy url
664     		 httpRequest.url = instance.getProxy();
665 
666     		 // Encode both url and parameters under the param url
667     		 params = [{ name: "url", value: sSource + '?' + jQuery.param(aoData) }];
668 
669     		 // Data type 
670     		 httpRequest.dataType = instance.getProxyDataType();
671     		 
672     	 }
673     	 
674     	// Wrap the callback function 
675 		 httpRequest.success = function ( data ) {
676 
677 			 // Decode data
678 			 jsonData = instance._decodeToJSON( data );
679 			 
680 			 // Set the echo var
681 			 jsonData.sEcho = this.sEcho;
682 			 
683 			 // Add the column of checkboxes in case of using row selection
684 			 instance._setSelectionColumn( jsonData );
685 			 
686 			 // Call the datatables cb function
687 			 fnCallback( jsonData );
688 
689 			 // fire event
690 			 instance.raiseEvent( Biojs.Table.EVT_ON_DATA_ARRIVED, { 
691 				 "jsonData": jsonData 
692 			 });
693 		 }
694 
695     	 httpRequest.type = 'GET';
696     	 httpRequest.data = params;
697     	 httpRequest.sEcho = aoData[0].value;
698 
699     	 jQuery.ajax( httpRequest );
700      },

730       * Function: Biojs.Table._decodeToJSON
731       * Purpose:  Decode the received data to suit it into the expected JSON format. 
732       * 		  Override this method in the Biojs.Table's children to suit any raw data into the expected JSON format.
733       * Returns:  {Object} formatted in the expected JSON format. 
734       * Inputs:   data -> {*} raw data received from the server. 
735       */
736      _decodeToJSON: function ( data ) {
737     	 var jsonData = data;
738     	 
739     	 if ( Biojs.Utils.isEmpty(data) ) {
740     		 Biojs.console.log("Empty data was received");
741     		 
742     	 } else if ( !(jsonData instanceof Object) || !(jsonData.aaData instanceof Array) ) {
743 			 jsonData = {};
744 			 jsonData.aaData = [];
745 			 jsonData.iTotalRecords = 0;
746 			 jsonData.iTotalDisplayRecords = 0;
747 			 Biojs.console.log("Error: data with unknown format was detected.");
748 		 }
749     	 
750     	 Biojs.console.log(jsonData);
751     	 return jsonData;
752      },

869     		 } else {

875     			 }
876     			 if ( columns[j].width && typeof columns[j].width == "string") {
877     				 result[j].sWidth = columns[j].width;
878     			 }
879     		 }
880     	 }
881     	 oSettings.aoColumns = result;
