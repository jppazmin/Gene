165   constructor: function (options) {
166 	  
167 	  var self = this;
168 	  var arrowType = this.opt.arrowType;
169 
170 	  this._container = jQuery('<div id="biojsTooltip' + self.getId() + '"></div>').addClass("Tooltip");
171 	  
172 	  this._arrow = jQuery('<div class="arrow"></div>').appendTo( self._container );
173 	  this._body = jQuery('<div class="body"></div>').appendTo( self._container );
174 	  
175 	  this._container.appendTo('body');
176 	  
177 	  this._initialize();

216   _initialize: function( ) {
217 
218 	  var self = this;
219       var timer = 0;
220       var targetSelector = this.opt.targetSelector;
221       var cbRender = this.opt.cbRender;
222       var refPos = this.opt.position;
223       var target;
224 
225       if ( this.opt.targetSelector instanceof jQuery ) {
226          targetSelector = this.opt.targetSelector;
227       } else {
228          targetSelector = jQuery( this.opt.targetSelector );
229       }
230 
231       // Get the text from the title attribute by default 
232       if ( "function" != typeof cbRender ) {
233     	  cbRender = function( element ) {
234     		  return jQuery( element ).attr('title');
235     	  }
236       }
237       
238       // Add the class to the arrow 
239       //arrow.addClass( self.opt.arrowType.match(/^(left|top|right|bottom)/g)[0] );
240       this.setArrowType( this.opt.arrowType );
241 
242       // Positioning 
243       this._arrow.css({ 
244     	  "position": "absolute", 
245     	  "z-index": "99999" 
246       });
247       
248       this._body.css({ 
249     	  'position': 'absolute', 
250     	  'z-index': '99998', 
251     	  'margin': '0px'
252       });
253       
254       if ( refPos == Biojs.Tooltip.MOUSE_POSITION ) {
255     	  targetSelector.mousemove( function (e) {
256     			
257         	  target = jQuery(e.target);
258         	  
259               if (timer) { 
260                   clearTimeout(timer); 
261               }
262               timer = 0;
263 
264               // Set the content
265               content = cbRender.call( self, e.target, self._body );
266 
267               //self._body.html( content );
268               // Show up
269               self._show();
270               
271               if ( content != undefined ) {
272             	  self.setContent( content );
273               }
274               
275               // Positioning 
276         	  self._setPosition(
277         			  { left: e.pageX - 10, top: e.pageY - 10 }, 
278         			  { width: 20, height: 20 } 
279         	  ); 
280               
281               // Event triggering 
282               self.raiseEvent( Biojs.Tooltip.EVT_ON_SHOW_UP, { 'target': target });
283           
284           }).mouseout( function() {
285               timer = setTimeout( 'Biojs.getInstance(' + self.getId()  + ')._hide()' , self.opt.delay );
286           });
287     	  
288       } else {
289     	  targetSelector.mouseenter( function (e) {
290     		  
291         	  target = jQuery(e.target);
292         	  
293               if (timer) { 
294                   clearTimeout(timer); 
295               }
296               timer = 0;
297              
298               // Show up
299               self._show();
300               
301               // Set the content
302               content = cbRender.call( self, e.target, self._body );
303 
304               //self._body.html( content );
305               if ( content != undefined ) {
306             	  self.setContent( content );
307               }
308 
309               // Positioning 
310               self._setPosition(
311             		  target.offset(), 
312             		  { width: target.width(), height: target.height() } 
313               );
314               
315               // Event triggering 
316               self.raiseEvent( Biojs.Tooltip.EVT_ON_SHOW_UP, { 'target': target });
317           
318           }).mouseout( function() {
319               timer = setTimeout( 'Biojs.getInstance(' + self.getId()  + ')._hide()' , self.opt.delay );
320           });
321       }
322       
323       self._container.mouseover( function(){
324           clearTimeout(timer);
325           timer = 0;
326           self._show();
327       
328       }).mouseout( function() {
329           timer = setTimeout( 'Biojs.getInstance(' + self.getId()  + ')._hide()' , self.opt.delay );
330       });
331       
332       this._hide();
333   },
334  
335   _hide: function() {
336 	  this._container.hide();
337   },
338   
339   _show: function() {
340 	  this._container.show();
341 	  this.raiseEvent( Biojs.Tooltip.EVT_ON_SHOW_UP, {} );

344   _setPosition: function ( offset, dim ) {
345 
346 	  var arrow = this._arrow;
347 	  var arrowType = this.opt.arrowType;
348 	  var arrowPos = { top: offset.top, left: offset.left };
349 
350 	  var body = this._body;
351 	  var bodyPos = {};
352 
353 	  arrow.removeClass();
354 	  arrow.addClass( 'arrow ' + arrowType.match(/^(left|top|right|bottom)/g)[0] );
355 	  
356 	  if ( arrowType == Biojs.Tooltip.ARROW_LEFT_TOP ) {
357 		  
358 		  arrowPos.top += Math.floor(dim.height/2) - Math.floor(arrow.height()/2);
359 		  arrowPos.left += arrow.width() + dim.width ;
360 		  
361 		  bodyPos.left = arrowPos.left + arrow.width() - 1;
362 		  bodyPos.top = arrowPos.top + Math.floor(arrow.height()/2) - Math.floor(body.height()/4);
363 		  
364 	  } else if ( arrowType == Biojs.Tooltip.ARROW_LEFT_MIDDLE ) {
365 
366 		  arrowPos.top += Math.floor(dim.height/2) - Math.floor(arrow.height()/2);
367 		  arrowPos.left += arrow.width() + dim.width ;
368 		  
369 		  bodyPos.left = arrowPos.left + arrow.width() - 1;
370 		  bodyPos.top = arrowPos.top + Math.floor(arrow.height()/2) - Math.floor(body.height()/2);
371 		  
372 	  } else if ( arrowType == Biojs.Tooltip.ARROW_LEFT_BOTTOM ) {
373 
374 		  arrowPos.top += Math.floor(dim.height/2) - Math.floor(arrow.height()/2);
375 		  arrowPos.left += arrow.width() + dim.width ;
376 		  
377 		  bodyPos.left = arrowPos.left + arrow.width() - 1;
378 		  bodyPos.top = arrowPos.top + Math.floor(arrow.height()/2) - Math.floor(body.height()*(3/4));
379 		  
380 	  } else if ( arrowType == Biojs.Tooltip.ARROW_TOP_LEFT ) {
381 		  
382 		  arrowPos.top += dim.height;
383 		  arrowPos.left += Math.floor(dim.width/2) - Math.floor(arrow.width()/2);
384 		  
385 		  bodyPos.left = arrowPos.left + Math.floor(arrow.height()/2) - Math.floor(body.width()/4);
386 		  bodyPos.top = arrowPos.top + arrow.height() -1;
387 		  
388 	  } else if ( arrowType == Biojs.Tooltip.ARROW_TOP_MIDDLE ) {
389 		  
390 		  arrowPos.top +=  dim.height;
391 		  arrowPos.left += Math.floor(dim.width/2) - Math.floor(arrow.width()/2);
392 		  
393 		  bodyPos.left = arrowPos.left + Math.floor(arrow.height()/2) - Math.floor(body.width()/2);
394 		  bodyPos.top = arrowPos.top + arrow.height() -1;
395 		  
396 	  } else if ( arrowType == Biojs.Tooltip.ARROW_TOP_RIGHT ) {
397 
398 		  arrowPos.top += dim.height;
399 		  arrowPos.left += Math.floor(dim.width/2) - Math.floor(arrow.width()/2);
400 		  
401 		  bodyPos.left = arrowPos.left + Math.floor(arrow.height()/2) - Math.floor( body.width()*(3/4) );
402 		  bodyPos.top = arrowPos.top + arrow.height() -1;
403 		  
404 	  } else if ( arrowType == Biojs.Tooltip.ARROW_RIGHT_TOP ) {
405 
406 		  arrowPos.top += Math.floor(dim.height/2) - Math.floor(arrow.height()/2);
407 		  arrowPos.left -= arrow.width();
408 		  
409 		  bodyPos.left = arrowPos.left - body.outerWidth() +1;
410 		  bodyPos.top = arrowPos.top + Math.floor(arrow.height()/2) - Math.floor(body.height()/4);
411 		  
412 	  } else if ( arrowType == Biojs.Tooltip.ARROW_RIGHT_MIDDLE ) {
413 		  
414 		  arrowPos.top += Math.floor(dim.height/2) - Math.floor(arrow.height()/2);
415 		  arrowPos.left -= arrow.width();
416 		  
417 		  bodyPos.left = arrowPos.left - body.outerWidth() +1;
418 		  bodyPos.top = arrowPos.top + Math.floor(arrow.height()/2) - Math.floor(body.height()/2);
419 		  
420 	  } else if ( arrowType == Biojs.Tooltip.ARROW_RIGHT_BOTTOM ) {
421 		  
422 		  arrowPos.top += Math.floor(dim.height/2) - Math.floor(arrow.height()/2);
423 		  arrowPos.left -= arrow.width();
424 		  
425 		  bodyPos.left = arrowPos.left - body.outerWidth() +1;
426 		  bodyPos.top = arrowPos.top + Math.floor(arrow.height()/2) - Math.floor(body.height()*(3/4));
427 		  
428 	  } else if ( arrowType == Biojs.Tooltip.ARROW_BOTTOM_LEFT ) {
429 		  
430 		  arrowPos.top -= arrow.height();
431 		  arrowPos.left += Math.floor(dim.width/2) + Math.floor(arrow.width()/2);
432 		  
433 		  bodyPos.left = arrowPos.left + Math.floor(arrow.width()/2) - Math.floor(body.width()/4);
434 		  bodyPos.top = arrowPos.top - body.outerHeight() +1;
435 
436 	  } else if ( arrowType == Biojs.Tooltip.ARROW_BOTTOM_MIDDLE ) {
437 		  
438 		  arrowPos.top -= arrow.height();
439 		  arrowPos.left += Math.floor(dim.width/2) - Math.floor(arrow.width()/2);
440 		  
441 		  bodyPos.left = arrowPos.left + Math.floor(arrow.width()/2) - Math.floor(body.width()/2);
442 		  bodyPos.top = arrowPos.top - body.outerHeight() +1;
443 
444 	  } else if ( arrowType == Biojs.Tooltip.ARROW_BOTTOM_RIGHT ) {
445 		  
446 		  arrowPos.top -= arrow.height();
447 		  arrowPos.left += Math.floor(dim.width/2) - Math.floor(arrow.width()/2);
448 		  
449 		  bodyPos.left = arrowPos.left + Math.floor(arrow.width()/2) - Math.floor(body.width()*(3/4));
450 		  bodyPos.top = arrowPos.top - body.outerHeight() +1;
451 	  }
452 
453 	  this._arrow.css( arrowPos );
454 	  this._body.css( bodyPos );
455   },

486 		  this.opt.arrowType = arrowType;
