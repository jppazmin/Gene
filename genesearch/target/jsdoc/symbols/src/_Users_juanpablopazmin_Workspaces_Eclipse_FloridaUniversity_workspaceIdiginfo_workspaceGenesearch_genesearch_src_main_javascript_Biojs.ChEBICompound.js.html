1 /**
  2  *  
  3  * 
  4  * @class
  5  * @extends Biojs
  6  * 
  7  * @requires <a href=''>Server side proxy</a>
  8  * 
  9  * @requires <a href='../biojs/css/ChEBICompound.css'>ChEBICompound.css</a>
 10  * @dependency <link href="../biojs/css/biojs.ChEBICompound.css" rel="stylesheet" type="text/css" />
 11  * 
 12  * @requires <a href='http://blog.jquery.com/2011/09/12/jquery-1-6-4-released/'>jQuery Core 1.6.4</a>
 13  * @dependency <script language="JavaScript" type="text/javascript" src="../biojs/dependencies/jquery/jquery-1.6.4.js"></script>
 14  * 
 15  * @param {Object} options An object with the options for the component.
 16  * 
 17  * @option {string} [imageUrl="http://www.ebi.ac.uk/chebi/displayImage.do"] 
 18  *    Url of the web service in order to require the compound image. 
 19  *    To get a compound image, 'imageUrl + id' will be used as URI.
 20  * 
 21  * @option {string} id
 22  *    ChEBI identifier of the compound to be displayed (i.e. '4991'). 
 23  *     
 24  * @option {int} [height=undefined] 
 25  *    The height in pixels of how big this component should be displayed. If it's not specified, the CSS value will be used instead. 
 26  * 
 27  * @option {int} [width=undefined]
 28  *    The width in pixels of how big this image should be displayed. If it's not specified, the CSS value will be used instead.
 29  * 
 30  * @example
 31  * var instance = new Biojs.ChEBICompound({
 32  *    target: 'YourOwnDivId',
 33  *    id: 'CHEBI:2922',
 34  *    width: 700,
 35  *    height: 400
 36  * });	
 37  * 
 38  */
 39 Biojs.ChEBICompound = Biojs.extend(
 40 /** @lends Biojs.ChEBICompound# */
 41 {
 42 	constructor: function(options){
 43 
 44 		var self = this;
 45 		var width = this.opt.width;
 46 		var height = this.opt.height;
 47 		var imageWidth, imageHeight;
 48 
 49 		if ( "string" == (typeof this.opt.target) ) {
 50 			this._container = jQuery( "#" + this.opt.target );
 51 		
 52 		} else {
 53 			
 54 			this.opt.target = "biojs_ChEBICompound_" + this.getId();
 55 			this._container = jQuery('<div id="'+ self.opt.target +'"></div>');
 56 		}
 57 
 58 		this._container.html('').addClass("ChEBICompound");
 59 		
 60 		if ( width == undefined ) {
 61 			width = this._container.css('width');
 62 			
 63 		} else {
 64 			this._container.width( width );
 65 		}
 66 		
 67 		if ( height == undefined ) {
 68 			height = this._container.css('height');
 69 			
 70 		} else {
 71 			this._container.height( height );
 72 		}
 73 
 74 		this._container.html('');
 75 		this._imageContainer = jQuery('<div class="ChEBICompound_image"></div>').appendTo(this._container);
 76 		this._tabContainer = jQuery('<div class="ChEBICompound_tab"></div>').appendTo(this._container);
 77 		
 78 		// Build the summary left panel 
 79 		this._summaryContainer = this._buildTabPanel(
 80 			this._tabContainer,
 81 			function ( tabVisible, visibleWidth ) {
 82 				// Do nothing on tab visibility change
 83 			}
 84 		);
 85 		
 86 		// Size of the image for the URL request  
 87 		imageWidth = this._imageContainer.width();
 88 		imageHeight = this._imageContainer.height();
 89 		this.opt.imageDimension = (imageWidth < imageHeight)? imageWidth: imageHeight;
 90 		
 91 		if (this.opt.id !== undefined) {
 92 			this.setId(this.opt.id);
 93 		}
 94 	},
 95 	
 96 	opt: {
 97 	   target: 'YourOwnDivId',
 98 	   id: undefined,
 99 	   imageUrl: 'http://www.ebi.ac.uk/chebi/displayImage.do',
100 	   chebiDetailsUrl: 'http://www.ebi.ac.uk/webservices/chebi/2.0/test/getCompleteEntity?chebiId=',
101 	   proxyUrl: '../biojs/dependencies/proxy/proxy.php',
102 	   
103 	   height: undefined,
104 	   width: undefined,
105 	   scale: false,
106 	   imageIndex: 0
107 	},
108 	
109 	eventTypes : [
110   		/**
111   		 * @name Biojs.ChEBICompound#onRequestError
112   		 * @event
113   		 * @param {function} actionPerformed An function which receives an {@link Biojs.Event} object as argument.
114   		 * @eventData {Object} source The component which did triggered the event.
115   		 * @eventData {string} file The name of the loaded file.
116   		 * @eventData {string} result A string with either value 'success' or 'failure'.
117   		 * @eventData {string} message Error message in case of result be 'failure'.
118   		 * 
119   		 * @example 
120   		 * instance.onRequestError(
121   		 *    function( e ) {
122   		 *       alert( e.message );
123   		 *    }
124   		 * ); 
125   		 * 
126   		 * */
127   		"onRequestError",
128   		/**
129   		 * @name Biojs.ChEBICompound#onImageLoaded
130   		 * @event
131   		 * @param {function} actionPerformed An function which receives an {@link Biojs.Event} object as argument.
132   		 * @eventData {Object} source The component which did triggered the event.
133   		 * @eventData {string} id The identifier of the loaded file.
134   		 * 
135   		 * @example 
136   		 * instance.onImageLoaded(
137   		 *    function( e ) {
138   		 *       alert( e.id + " loaded." );
139   		 *    }
140   		 * ); 
141   		 * 
142   		 * */
143   		"onImageLoaded"
144 	],
145 
146 	/**
147     * Set the identifier of the chemical component. 
148     * Shows both information and image for the new identifier.
149     * 
150     * @param {string} chebiId Chemical EBI's identifier for the compound to be displayed.
151     *
152     * @example
153     * instance.setId('CHEBI:4991');
154     * 
155     * @example
156     * // No image available 
157     * instance.setId('CHEBI:60004');
158     * 
159     */
160 	setId: function( chebiId ) {
161 		
162 		var self = this;
163 		var url;
164 		var image;
165 		
166 		this._imageContainer.html('');
167 		this.opt.id = chebiId.replace('CHEBI:','');
168 		this._requestDetails( this.opt );
169 		
170 		var params = { 
171 			defaultImage:	true, 
172 			imageIndex: 	this.opt.imageIndex,
173 			chebiId: 		chebiId,
174 			dimensions: 	this.opt.imageDimension,
175 			scaleMolecule: 	this.opt.scale
176 		};
177 		
178 		url = this.opt.imageUrl + '?' + jQuery.param(params);
179 		image = jQuery('<img id="image_' + chebiId + '"/>');
180 		
181 		image.attr('src', url)
182         	.load(function() {
183 	           if (!this.complete || typeof this.naturalWidth == "undefined" || this.naturalWidth == 0) {
184 	        	   self._imageContainer.addClass("noImage");
185 	        	   self.raiseEvent( Biojs.ChEBICompound.EVT_ON_REQUEST_ERROR, {
186 	        		   id: chebiId,
187 	        		   url: url,
188 	        		   message: "No image available"
189 	   			   });
190 
191 	           } else {
192 	        	   self._imageContainer.removeClass("noImage");
193 	        	   self._imageContainer.append(image).css({
194 	           			'width': self.opt.imageDimension,
195 	           			'height': self.opt.imageDimension,
196 	           			'margin': 'auto'
197 	        	   });
198 	        	   
199 	        	   self.raiseEvent( Biojs.ChEBICompound.EVT_ON_IMAGE_LOADED, {
200 	        		   id: chebiId,
201 	        		   url: url
202 	   			   });
203 	           }
204         	})
205         	
206 
207 	},
208 	
209 	_requestDetails: function( opt ){
210 		var self = this;
211 		
212 		var urlSummary = opt.chebiDetailsUrl + opt.id;
213 		
214 		Biojs.console.log( "Requesting summary from: " + urlSummary );
215 		
216 		var httpRequest = {
217 			url: urlSummary,
218 			method: "GET",
219 			/**
220 			 * @ignore
221 			 */
222 			success: function(xml){
223 				self._dataReceived(xml);
224 			},
225 			/**
226 			 * @ignore
227 			 */
228 			error: function(qXHR, textStatus, errorThrown) {
229 				Biojs.console.log("ERROR requesting summary. Response: " + textStatus);
230 				self.raiseEvent( Biojs.ChEBICompound.EVT_ON_REQUEST_ERROR, {
231 					message: textStatus
232 				});
233 			}
234 		};
235 		
236 		// Using proxy?
237 	   	// Redirect using the proxy and encode all params as url data
238 	   	if ( opt.proxyUrl != undefined ) {
239 	
240 	   		 // Redirect to proxy url
241 	   		 httpRequest.url = opt.proxyUrl;
242 	
243 	   		 // Encode both url and parameters under the param url
244 	   		 httpRequest.data = [{ name: "url", value: urlSummary }];
245 	
246 	   		 // Data type 
247 	   		 httpRequest.dataType = "text";
248 	   	}
249 
250 		jQuery.ajax(httpRequest);
251 		
252 	},
253 	
254 	
255 	// parses the xml file from the request and stores the information in an easy to access way
256 	_dataReceived: function(xml){
257 	    var data = {};
258 	    var i = 0;
259 		var self = this;
260 		
261 		Biojs.console.log("Data received");
262 		
263 		if ( xml.length > 0 ) {
264 			xmlDoc = jQuery.parseXML( xml );
265 		    xmlResult = jQuery(xmlDoc).find('return');
266 			
267 		    data.chebiAsciiName = { name: "Name", value: xmlResult.find(' > chebiAsciiName').text() };
268 			data.chebiId = { name: "Identifier", value: xmlResult.find('> chebiId').text() };
269 			data.definition = { name: "Definition", value: xmlResult.find('> definition').text() };
270 			data.SecondaryChEBIIds = { name: "Other Identifiers", value: xmlResult.find(' > SecondaryChEBIIds').text() };
271 			data.entityStar = { name: "Stars", value: xmlResult.find(' > entityStar').text() };
272 			
273 		}
274 
275 		this._setSummary( data );
276 		
277 	    return data;
278 	},
279 
280 	
281 	_setSummary : function ( data ) {
282 		Biojs.console.log("_setSummary()");
283 
284 		var container = this._summaryContainer;
285 		
286 		// Remove all elements in container 
287 		container.children().remove();
288 		
289 		if ( Biojs.Utils.isEmpty(data) ) {
290 			container.append('Not information available');
291 			
292 		} else {
293 			 // Add the summary data
294 			for (key in data) {
295 				if ( data[key].value.length > 0 ) {
296 					if ( key == 'entityStar' ) {
297 						jQuery('<h2>' + data[key].name + '</h2>').appendTo( container );
298 						jQuery('<div class="star"/>')
299 							.appendTo( container )
300 							.css({
301 								'width': parseInt(data[key].value) * 16,
302 								'padding': 0
303 							});
304 					} else {
305 						container.append( '<h2>' + data[key].name + '</h2><p>' + data[key].value + '</p>' );
306 					}
307 				}
308 			}
309 			
310 			this.raiseEvent( Biojs.ChEBICompound.EVT_ON_SUMMARY_LOADED,{
311 				id: data.Identifier
312 			});
313 		}
314 
315 		Biojs.console.log("_setSummary done");
316 	},
317 	
318 	_buildTabPanel: function( container, onVisibilityChangeCb ) {
319 		
320 		container.html('');	
321 		
322 		var content = jQuery('<div class="content"></div>').appendTo(container);
323 		var expand = jQuery('<div style="display: none;" class="toggle expand"></div>').appendTo(container);		
324 		var collapse = jQuery('<div class="toggle collapse"/>').appendTo(container);
325 		
326 		var buttonsWidth = parseInt( jQuery('.toggle').css('width'), 10 );
327 		var contentWidth = container.width() - buttonsWidth;
328 		
329 		container.css( 'left', 0 )
330 			.find('.toggle')
331 			.click( function(){
332 				// Animate show/hide this tab
333 				container.animate({ 
334 						left: ( parseInt( container.css('left'), 10 ) == 0 ? buttonsWidth - container.outerWidth() : 0 ) + "px"
335 					},
336 					// to call once the animation is complete 
337 					function() {
338 						
339 						var visibleWidth = parseInt( container.css('left'), 10 ) == 0 ? container.width() : buttonsWidth;
340 						
341 						container.find('.toggle').toggle();
342 
343 						if ( "function" == typeof onVisibilityChangeCb ) {
344 							onVisibilityChangeCb.call( this, collapse.is(':visible'), visibleWidth );
345 						}
346 					}
347 				);
348 			})
349 			.css({
350 				'float':'right',
351 				'position':'relative'
352 			});
353 		
354 		Biojs.console.log('content width '+ contentWidth);
355 		
356 		content.css({ 
357 			'width': contentWidth + "px", 
358 			'height': container.height() + "px", 
359 			'word-wrap': 'break-word'
360 		});
361 		
362 		return content;
363 	},
364 	
365 	getHTML: function () {
366 		return this._container;
367 	}
368 	
369 },{
370 	//Events 
371 	EVT_ON_IMAGE_LOADED: "onImageLoaded",
372 	EVT_ON_SUMMARY_LOADED: "onSummaryLoaded",
373 	EVT_ON_REQUEST_ERROR: "onRequestError"
374 	
375 });
