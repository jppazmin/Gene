 15  * @param {Object} options An object with the options for Sequence component.
 16  *    
 17  * @option {string} target 
 18  *    Identifier of the DIV tag where the component should be displayed.
 19  * 
 20  * @option {string} identifier 
 21  * 	  ENSEMBL gene identifier or UniProt Acc needed as input to fetch the Gene expression summary data

 32  *    You can use tour own proxy script by modifying this value. 
 33  * 
 34  * @example
 35  * var instance = new Biojs.GeneExpressionSummary({
 36  * 	  target: 'YourOwnDivId',
 37  * 	  identifier: 'ENSG00000139618'
 38  * });
 39  * 
 40  */
 41 Biojs.GeneExpressionSummary = Biojs.extend(
 42 /** @lends Biojs.GeneExpressionSummary# */
 43 {	
 44 	constructor: function (options) {
 45 		//Biojs.console.enable();
 46 		this.setIdentifier(this.opt.identifier);	
 47 	},

 76 		 * @example 
 77 		 * instance.onRequestError(
 78 		 *    function( e ) {
 79 		 *       alert( e.message );
 80 		 *    }
 81 		 * ); 
 82 		 * 
 83 		 **/
 84 		"onRequestError",
 85 		/**
 86 		 * @name Biojs.GeneExpressionSummary#onDbError
 87 		 * @event
 88 		 * @param {function} actionPerformed A function which receives an {@link Biojs.Event} object as argument.
 89 		 * @eventData {Object} source The component which did triggered the event.
 90 		 * @eventData {string} file The name of the loaded file.
 91 		 * @eventData {string} result A string with either value 'success' or 'failure'.
 92 		 * @eventData {string} message Error message in case of result be 'failure'.
 93 		 * 
 94 		 * @example 
 95 		 * instance.onDbError(
 96 		 *    function( e ) {
 97 		 *       alert( e.message );
 98 		 *    }
 99 		 * ); 
100 		 * 
101 		 **/
102 		"onDbError"
103 	],
104 	/**
105 	 * Fetch the data by means of identifier.
106 	 * @param {string} identifier The segment identifier.
107 	 * 
108 	 * @example 
109 	 * instance.setIdentifier("ENSG00000100867");
110 	 * 
111 	 * @example 
112 	 * instance.setIdentifier("Q61171");
113 	 * 
114 	 * @example 
115 	 * instance.setIdentifier("ENSG000000000000");
116 	 * 
117 	 * @example 
118 	 * instance.setIdentifier("P00000");
119 	 * 
120 	 */
121 	setIdentifier: function(id){
122 		this._drawTemplate();
123 		/* Uniprot or Ensembl ID? */
124 		this._identifierDb = this._checkIdentifier(id);
125 		if(this._identifierDb == Biojs.GeneExpressionSummary.ID_UNIPROT){
126 			this._requestFeaturesFromUniprotAcc(id);
127 		} else if (this._identifierDb == Biojs.GeneExpressionSummary.ID_ENSEMBL){
128 			this._requestFeatures(id);	
129 		} else {
130 			this._processDbError(id);
131 		}	
132 	},
133 	/* 
134      * Function: Biojs.GeneExpressionSummary._drawTemplate
135      * Purpose:  Draw HTML template used for placing the data afterwards
136      * Returns:  -
137      * Inputs:  -
138      */
139 	_drawTemplate: function(){
140 		var self = this;
141 		
142 		this._selector = "#" + this.opt.target;
143 		this._container = jQuery(this._selector);
144 		this._container.html("");
145 		
146 		this._container.addClass("GeneExpressionSummary");
147 		
148 		this._titleContainer = jQuery('<div class="GeneExpressionSummary_title"></div>').appendTo(this._container);
149 		
150 		this._title = jQuery('<span>Loading...</span>').appendTo(this._titleContainer);
151 		
152 		this._compactButton = jQuery('<span style="float:right;cursor:pointer;" class="toggle collapse"></span>')
153 			.appendTo(this._titleContainer)
154 			.click(function(e){
155 				self._table.hide();
156 				self._footerContainer.hide();
157 				self._titleContainer.children('span.toggle').toggle();
158 				self.opt.collapsed = true;
159 			});
160 		
161 		this._expandButton = jQuery('<span style="float: right; display:none; cursor:pointer;" class="toggle expand"></span>')
162 			.appendTo(this._titleContainer)
163 			.click(function(e){
164 				self._table.show();
165 				self._footerContainer.show();
166 				self._titleContainer.children('span.toggle').toggle();
167 				self.opt.collapsed = false;
168 			});
169 		
170 		this._table = jQuery('<table class="GeneExpressionSummary_table"></table>').appendTo(this._container);
171 		this._row = jQuery("<tr></tr>").appendTo(this._table);
172 		this._leftColumn = jQuery("<td></td>").appendTo(this._row);
173 		this._rightColumn = jQuery("<td></td>").appendTo(this._row);
174 		
175 		this._imageContainer = jQuery('<div class="GeneExpressionSummary_image"></div>').appendTo(this._rightColumn);
176 		this._summaryContainer = jQuery('<div class="GeneExpressionSummary_summary"></div>').appendTo(this._leftColumn);
177 		this._footerContainer = jQuery('<div class="GeneExpressionSummary_footer"></div>').appendTo(this._container);
178 		
179 		if ( this.opt.collapsed ) {
180 			this._table.hide();
181 			this._footerContainer.hide();
182 			this._titleContainer.children('span.expand').show();
183 			this._titleContainer.children('span.collapse').hide();
184 		}
185 		
186 	},

196 			this._compactButton.click();

201      * Expands the view of the summary

208 			this._expandButton.click();
209 		}

212 	/* 
213      * Function: Biojs.GeneExpressionSummary._checkIdentifier
214      * Purpose:  Check if the indetifier provided by the user is ENSEMBL or UniProt
215      * Returns:  Database name (uniprot or ensembl)
216      * Inputs:   id -> {String} Identifier.
217      */
218 	_checkIdentifier: function(id){
219 		var self = this;
220 		self._re = /^([A-N,R-Z][0-9][A-Z][A-Z, 0-9][A-Z, 0-9][0-9])|([O,P,Q][0-9][A-Z, 0-9][A-Z, 0-9][A-Z, 0-9][0-9])$/;
221 		if (id.search(self._re) != -1){
222 			return Biojs.GeneExpressionSummary.ID_UNIPROT;
223 		} else if(id.substring(0,4) == "ENSG"){
224 			return Biojs.GeneExpressionSummary.ID_ENSEMBL;
225 		}	
226 	},
227 	/* 
228      * Function: Biojs.GeneExpressionSummary._requestFeatures
229      * Purpose:  Start a request using an ENSEMBL identifier
230      * Returns:  -
231      * Inputs:   id -> {String} Identifier.
232      */
233 	_requestFeatures: function( identifier ) {
234 		if ( undefined !== identifier) {
235 			if (identifier.length > 0) {
236 				this.opt.identifier = identifier;
237 				this._requestFeaturesXML(this.opt);
238 			} else {
239 				Biojs.console.log("no identifier value available");
240 				this._displayNoDataMessage();	
241 			}
242 		} else {
243 			Biojs.console.log("identifier value is not valid");
244 			this._displayNoDataMessage();
245 		}
246 	},
247 	
248 	/* 
249      * Function: Biojs.GeneExpressionSummary._requestFeaturesXML
250      * Purpose:  Request data to the server
251      * Returns:  -
252      * Inputs:   opt -> {Object} options object.
253      */
254 	_requestFeaturesXML: function( opt ){
255 		var self = this;
256 		var httpRequest = {
257 			url: opt.featuresUrl,
258 			data: "segment=" + opt.identifier,
259 			methid: "GET",
260 			/** @ignore No need to document this object */
261 			success: function(xml){
262 				Biojs.console.log("SUCCESS: data received");
263 				self._responseReceived(xml);
264 			},
265 			error: self._processErrorRequest
266 		};
267 		
268 		// Using proxy?
269 	   	// Redirect using the proxy and encode all params as url data
270 	   	if ( opt.proxyUrl != undefined ) {
271 	
272 	   		 // Redirect to proxy url
273 	   		 httpRequest.url = opt.proxyUrl;
274 	
275 	   		 // Encode both url and parameters under the param url
276 	   		 httpRequest.data = [{ name: "url", value: opt.featuresUrl + "?segment="+opt.identifier }];
277 	
278 	   		 // Data type 
279 	   		 httpRequest.dataType = "text";
280 	   	}
281 		
282 		jQuery.ajax(httpRequest);
283 		
284 	},
285 	
286 	/* 
287      * Function: Biojs.GeneExpressionSummary._responseReceived
288      * Purpose:  Parses the xml file from the request and stores the information in an easy to access way
289      * Returns:  {object[]} -> decoded features
290      * Inputs:   xml -> {string} xml with the features.
291      */
292 	_responseReceived: function( xml ){
293 		var self = this;
294 		var xmlDoc = "";
295 		var features = [];
296 		
297 		Biojs.console.log("Decoding " + xml);
298 		
299 		try {
300 			xmlDoc = jQuery.parseXML( xml );
301 			
302 		} catch (e) {
303 			Biojs.console.log("ERROR decoding ");
304 			Biojs.console.log(e);
305 		}
306 		
307 	    jQuery(xmlDoc).find('FEATURE')
308 	    	.each( function(){
309 	    		features.push( self._decodeFeature(this) );
310 	    	});
311 	    
312 		if (features.length > 0) {
313 			Biojs.console.log("Features decoded:");
314 			Biojs.console.log(features);
315 			this._setFeatures(features);
316 		} else {
317 			this._displayNoDataMessage();
318 		}
319 	},
320 
321 	_displayNoDataMessage: function(){
322 		jQuery('#'+this.opt.target+'').html(Biojs.GeneExpressionSummary.MESSAGE_NODATA);
323 	},
324 	
325 	/* 
326      * Function: Biojs.GeneExpressionSummary._requestFeaturesXML
327      * Purpose:  Convert a feature from XML to js object 
328      * Returns:  {object} js object containing the data of the feature 
329      * Inputs:   featureNode -> {Node} The DOM Node
330      */
331 	_decodeFeature: function ( featureNode ) {
332 		
333 	    Biojs.console.log("Decoding feature" + featureNode );
334 		
335 		var feature = this._decodeNode( featureNode );
336 	    
337 	    // Get the children of the feature node
338 	    var children = jQuery( featureNode ).children();
339 	    
340 	    // Decode each child 
341 	    for ( i = 0; i < children.length; i++ ) {
342 	    	var objChild = this._decodeNode(children[i]);
343 	    	
344 	    	// Add child to the feature
345 	    	if ( "NOTE" == children[i].nodeName ) {
346 	    		if (feature[ children[i].nodeName ] == undefined ) feature[ children[i].nodeName ] = []; 
347 	    		feature[ children[i].nodeName ].push( objChild );
348 	    	} else {
349 	    		feature[ children[i].nodeName ] = objChild;
350 	    	}
351 	    }
352 	    
353 	    return feature;
354 	},
355 	
356 	/* 
357      * Function: Biojs.GeneExpressionSummary._decodeNode
358      * Purpose:  Convert a node to a js object 
359      * Returns:  {object} js object containing the data of the node 
360      * Inputs:   node -> {Node} The DOM Node
361      */
362 	_decodeNode: function (node) {
363 		var obj = {}
364 
365     	// Get attributes
366     	jQuery.each( node.attributes, function(index, attr) {
367             obj[ attr.nodeName ] = attr.nodeValue;
368 	    });
369     	obj["text"] = jQuery(node).text();
370     	
371     	return obj;
372 	},

374 	/* 
375      * Function: Biojs.GeneExpressionSummary._setFeatures
376      * Purpose:  Build the HTML in this container using the provided features object.
377      * Returns:  -
378      * Inputs:   f -> {object} Features to be displayed
379      */
380 	_setFeatures: function ( f ) {
381 		
382 		// Flush the containers
383 		this._title.text('');
384 		this._imageContainer.text('');
385 		this._summaryContainer.text('');
386 		this._footerContainer.text('');
387 		
388 		var self = this;
389 		for ( i = 0; i < f.length; i++ ) {
390 			var feature = f[i];
391 			if ( Biojs.GeneExpressionSummary.TYPE_DESCRIPTION == feature.TYPE.id ) {
392 				this._title.text( feature.NOTE[0].text );
393 				
394 			} else if ( Biojs.GeneExpressionSummary.TYPE_IMAGE == feature.TYPE.id ) {
395 				var image = jQuery('<img src="'+ feature.LINK.href +'" />').appendTo(this._imageContainer);
396 				image.css("width","100%");
397 				image.css("height","auto");
398 				var caption = '<p>'+ feature.LINK.text;
399 				this._imageContainer.append(caption);
400 			} else if ( Biojs.GeneExpressionSummary.TYPE_PROVENANCE == feature.TYPE.id && self.opt.legend == true) {
401 				var footer = '';
402 				for ( n in feature.NOTE ) {
403 					footer += feature.NOTE[n].text + ' ';
404 				}
405 				footer += '<a href="' + feature.LINK.href + '">'+ feature.LINK.href + '</a>';
406 				this._footerContainer.append(footer);
407 				
408 			} else {
409 				var summary = '<div class="GeneExpressionSummary_subtitle">' + feature.label + '</div>';
410 				summary += '<div class="GeneExpressionSummary_feature">' + feature.NOTE[0].text;
411 				if(typeof feature.LINK != 'undefined'){
412 					summary += ' <a href="' + feature.LINK.href + '">view all</a></div>';	
413 				}
414 				this._summaryContainer.append(summary);
415 			}
416 			
417 		}
418 	},
419 	/* 
420      * Function: Biojs.GeneExpressionSummary._requestFeaturesFromUniprotAcc
421      * Purpose:  Start a request using a UniProt identifier
422      * Returns:  -
423      * Inputs:   id -> {String} UniProt identifier.
424      */
425 	_requestFeaturesFromUniprotAcc: function(id){
426 		var self = this;
427 		/* URL where to get the mapping to an EMSEMBL id */	
428 		self._serviceUrl = "http://www.ebi.ac.uk/uniprot/biomart/martservice?query=%3C?xml%20version=%221.0%22%20encoding=%22UTF-8%22?%3E%3C!DOCTYPE%20Query%3E%3CQuery%20%20virtualSchemaName%20=%20%22default%22%20formatter%20=%20%22TSV%22%20header%20=%20%220%22%20uniqueRows%20=%20%221%22%20count%20=%20%22%22%20datasetConfigVersion%20=%20%220.6%22%20%3E%3CDataset%20name%20=%20%22uniprot%22%20interface%20=%20%22default%22%20%3E%3CFilter%20name%20=%20%22accession%22%20value%20=%20%22"
429 		self._serviceUrl += id;
430 		self._serviceUrl += "%22/%3E%3CAttribute%20name%20=%20%22ensembl_id%22%20/%3E%3C/Dataset%3E%3C/Query%3E";
431 		self._url = "";
432 		if(self.opt.proxyUrl != ""){
433 			self._url= self.opt.proxyUrl + "?url=" + self._serviceUrl;
434 		} else {
435 			self._url= self._serviceUrl;	
436 		}
437 		
438 		/* process service */
439 	    jQuery.ajax({
440 		    type: "GET",
441 		    url: self._url,
442 			dataType: "text",
443 		    success: function(a){self._requestFeatures(a);},
444 			error: function(a){self._processErrorRequest(a);}
445 	    });
446 	},
447 	/* 
448      * Function: Biojs.GeneExpressionSummary._processErrorRequest
449      * Purpose:  Process request error
450      * Returns:  -
451      * Inputs:   textStatus -> {String} Text status
452      */
453 	_processErrorRequest: function (textStatus){
454 		Biojs.console.log("ERROR: " + textStatus );
455 		this.raiseEvent( Biojs.GeneExpressionSummary.EVT_ON_REQUEST_ERROR, { message: textStatus } );
456 	},
457 	/* 
458      * Function: Biojs.GeneExpressionSummary._processDbError
459      * Purpose:  Process DB error
460      * Returns:  -
461      * Inputs:   id -> {String} Identifier.
462      */
463 	_processDbError: function (id){
464 		var self = this;
465 		self._message = "Not recognize identifier: " + id;
466 		Biojs.console.log("ERROR: " + self._message );
467 		self.raiseEvent( Biojs.GeneExpressionSummary.EVT_ON_DB_ERROR, { message: self._message } );
468 	}
