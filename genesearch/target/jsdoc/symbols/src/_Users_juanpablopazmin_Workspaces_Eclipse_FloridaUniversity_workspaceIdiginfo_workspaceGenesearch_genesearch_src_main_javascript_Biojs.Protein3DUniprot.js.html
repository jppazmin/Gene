1 /**
  2  *
  3  * Protein viewer to show the pdb files aligned to an Uniprot Accession
  4  *
  5  * @class
  6  * @extends Biojs.Protein3DWS
  7  *
  8  * @param {Object} options An object with the options for the component.
  9  *
 10  * @option {string} [proteinId]
 11  *    Uniprot identifier of the protein.
 12  *
 13  * @option {string} [alignmentsUrl="http://www.rcsb.org/pdb/rest/das/pdb_uniprot_mapping/alignment?query="]
 14  *    Url to obtain the protein's alignments.
 15  *
 16  * @example
 17  *
 18  * var instance = new Biojs.Protein3DUniprot({
 19  * 		target: 'YourOwnDivId',
 20  * 		proteinId: 'P07148'
 21  * });
 22  *
 23  */
 24 Biojs.Protein3DUniprot = Biojs.Protein3DWS.extend(
 25 /** @lends Biojs.Protein3DUniprot# */

 27 	constructor: function(options) {
 28 		this.base(options);
 29         var self = this;
 30         
 31         this.onPdbLoaded(function(e) {
 32             Biojs.console.log(e.result + " loading the pdb file " + e.file);
 33             Biojs.console.log("self._aligmentsJustArrived= " + self._alignmentsJustArrived);
 34             if (self._alignmentsJustArrived) {
 35                 Biojs.console.log("Initialising the alignments selection list");
 36                 self.reset();
 37                 var alignments = self._filterAligmentsBySelection(self._selection);
 38                 var pdbOptions = self._createOptions(alignments);
 39                 if (jQuery('#' + self.opt.target).find('div#pdbStructures').length == 0) {
 40                     self._addControl('<div id="pdbStructures"></div>');
 41                 }
 42                 jQuery('#' + self.opt.target).find('div#pdbStructures').html('<h1>Structures for <b>' + self.opt.proteinId + '</b></h1><br/>' + '<select id="pdbFile_select">' + pdbOptions + '</select>');
 43                 jQuery('#' + self.opt.target + ' #pdbFile_select').val(pdb);
 44                 jQuery('#' + self.opt.target + ' #pdbFile_select').change(function() {
 45                     self._onAlignmentSelectionChange();
 46                 });
 47                 jQuery('#' + self.opt.target).find('#pdbStructures').show();
 48                 var alignmentId = pdb;
 49                 var pdbId = alignmentId.substring(0, pdb.indexOf('.')).toLowerCase();
 50                 var alignment = self.getAlignmentsByPdb(alignmentId);
 51                 if (alignment.hasOwnProperty(alignmentId)) {
 52                     var start = alignment[alignmentId][1].start;
 53                     var end = alignment[alignmentId][1].end;
 54                     self.raiseEvent('onPdbSelected', {"pdbId": pdbId,"alignmentId": alignmentId,"start": start,"end": end});
 55                 }
 56                 self._alignmentsJustArrived = false;
 57             }
 58         });
 59  
 60         if (this.opt.proteinId != undefined) {
 61             var proteinId = this.opt.proteinId;
 62             this.opt.proteinId = '';
 63             this.setProtein(proteinId);
 64         }
 65 	},

 97 	/**
 98     * Get all pdb files for a given uniprot id.
 99     * Also triggers the event that a new pdb file was loaded.
100     *
101     * @option {string} proteinId Uniprot identifier of the protein.
102     *
103     * @example 
104     * instance.setProtein("P99999");
105     * 
106     */
107 	setProtein: function(proteinId){
108 		if (proteinId != this.opt.proteinId) {
109 			this.opt.proteinId = proteinId;
110 			this._selection = undefined;
111 			this._minStart = Number.MAX_VALUE;
112 			this._maxEnd = 0;
113 			this._alignments = undefined;
114 			this._requestAligmentsXML();
115 		} 
116 	},

118 	// makes an ajax request to get the pdb files for the given uniprot id
119 	_requestAligmentsXML: function(){
120 		var self = this;
121 
122 		self.showLoadingImage();
123 
124 		jQuery.ajax({
125 			url: this.opt.proxyUrl,
126 			data: 'url=' + self.opt.alignmentsUrl + self.opt.proteinId,
127 			dataType: "text",
128 			success: function(xml){
129 				Biojs.console.log("SUCCESS: data received");
130 				self._parseResponse(xml);
131 			},
132 			async: false,
133 			error: function(qXHR, textStatus, errorThrown){
134 				Biojs.console.log("ERROR: requesting "+this.data);
135 				self.raiseEvent('onRequestError', {message: textStatus});
136 			}
137 		});
138 	},

141 	// parses the xml file from the request and stores the information in an easy to access way
142 	_parseResponse: function(xml){
143 	    this._alignments = {};
144 	    var i = 0;
145 		var self = this;
146 		
147 		Biojs.console.log("Decoding " + xml);
148 		
149 		try {
150 			xmlDoc = jQuery.parseXML( xml );
151 			
152 			jQuery(xmlDoc).find('block').each( function(){
153 		        var children = jQuery(this).children();
154 		        var segment0 = self._createNode(children[0]);
155 		        var segment1 = self._createNode(children[1]);
156 		        
157 		        var arr = [];
158 		        arr.push(segment0);
159 		        arr.push(segment1);
160 		        
161 		        self._alignments[segment0.intObjectId] = arr || [];
162 		        i++;
163 				
164 			    if (self._minStart > segment1.start) {
165 			        self._minStart = segment1.start;
166 			    }
167 			    
168 			    if (self._maxEnd < segment1.end) {
169 			        self._maxEnd = segment1.end;
170 			    }
171 		    });
172 			
173 		} catch (e) {
174 			Biojs.console.log("Error decoding response: " + e.message );
175 			this._alignments = {};
176 		}
177 	    
178 	    Biojs.console.log("Alignments decoded:");
179 	    Biojs.console.log(self._alignments);
180 	    this._aligmentsArrived();
181 	},

183 	// creates a node for the data structure that holds the xml information
184 	_createNode: function(segment){
185 		var start = parseInt(jQuery(segment).attr('start'));
186 	    var end = parseInt(jQuery(segment).attr('end'));
187 	    var obj = {
188 	        intObjectId: jQuery(segment).attr('intObjectId') || '',
189 	        start: start || '',
190 	        end: end || ''
191 	    }
192 	    return obj;
193 	},

195 	// adds all available pdb files to a dropdown box 
196 	// or displays the fact that there are no pdb files for the given id
197 	_aligmentsArrived: function(){
198 		
199 		this._alignmentsJustArrived = true;
200         
201 		
202 		var alignments = this._filterAligmentsBySelection(this._selection);
203         
204         if (!Biojs.Utils.isEmpty(alignments)) {
205             var pdb = undefined;
206             for (pdb in alignments) {
207                 break;
208             }
209             
210             Biojs.console.log("Requesting pdb " + pdb);
211             this.requestPdb(pdb.substring(0, pdb.indexOf('.')).toLowerCase());
212             
213         } else {
214             this._container.html("No structural information for " + this.opt.proteinId );
215             this.raiseEvent('onRequestError', { message: "No structural information available for " + this.opt.proteinId });
216         }
217 	},

219 	_onAlignmentSelectionChange: function(){
220 		var pdb = jQuery('#pdbFile_select').val();
221 		
222 		if ( pdb != undefined ) {
223 			
224 			var alignmentId = pdb.substring(0, pdb.indexOf(' '));
225 			var pdbId = alignmentId.substring(0, pdb.indexOf('.')).toLowerCase();
226 			
227 			var alignment = this.getAlignmentsByPdb(alignmentId);
228 			
229 			if ( alignment.hasOwnProperty( alignmentId ) ) {
230 				var start = alignment[alignmentId][1].start;
231 				var end = alignment[alignmentId][1].end;
232 				
233 				this.raiseEvent('onPdbSelected', { 
234 					"pdbId": pdbId,
235 					"alignmentId": alignmentId,
236 					"start": start,
237 					"end": end
238 				});
239 			}
240 			
241 			this.requestPdb(pdbId);
242 			
243 		} else {
244 			Biojs.console.log("No structural information available for "+this.opt.proteinId);
245 		}
246 	},

248 	/**
249     * Get the available alignments for the current protein filtered by selection (PDB files containing a part of the requested region).
250     * 
251     * @example 
252     * // Selection of the region in the interval [100,150].
253     * instance.setSelection({start: 120, end: 150});
254     * 
255     * @example
256     * // Get the alignments matching with bases 4, 8 and 100.
257     * alert ( instance.filterAlignments([4,8,100]) );
258     * 
259     * @param {Object|Array} selection Can be either a plain object or an array.  
260     *        If object, it must have the fields start and end; Where "start" is greater than or equal to "end".
261     *        If array, it must contain numbers representing the positions to be selected.
262     */ 
263 	getAlignmentsBySelection: function ( selection ) {
264 		
265 		var alignments = this._alignments;
266 		
267 		if ( selection != undefined ) {
268 			this._filterAligmentsBySelection(selection);
269 		}
270 		
271 		return alignments;
272 	},

274 	/**
275     * Get the available alignments for the current protein filtered by selection (PDB files containing a part of the requested region).
276     * 
277     * @example
278     * // Get the alignments matching with bases 4, 8 and 100.
279     * alert ( instance.getAlignmentsByPdb("2F73") );
280     * 
281     * @param {string} pdb identifier.  
282     * @returns {Object} .  
283     * 
284     */ 
285 	getAlignmentsByPdb: function ( pdbId ) {
286 		
287 		var alignments = {};
288 		
289 		for (al in this._alignments) {
290 			if (  this._alignments[al][0].intObjectId.indexOf( pdbId ) != -1 ) {
291 				alignments[this._alignments[al][0].intObjectId] = this._alignments[al];
292 			}
293 		}
294 		
295 		Biojs.console.log("Alignments for pdb " + pdbId );
296 		Biojs.console.log(alignments);
297 		
298 		return alignments;
299 	},

301 	/**
302     * Filters the alignments available for the current protein: Only PDB files containing a part of the requested region
303     * are selectable. The specified region is highlighted in the displayed PDB file.
304     *
305     * @example 
306     * // Selection of the region in the interval [100,150].
307     * instance.filterAlignments({start: 120, end: 150});
308     * 
309     * @param {Object|Array} selection Can be either a plain object or an array.  
310     *        If object, it must have the fields start and end; Where "start" is greater than or equal to "end".
311     *        If array, it must contain numbers representing the positions to be selected.
312     */ 
313 	filterAlignments: function ( selection ) { 
314 		
315 		var alignments = this._filterAligmentsBySelection(selection);
316 		var selectedAlignment = jQuery('#pdbFile_select').val();
317 		
318 		// Update the drop-down box showing the filtered alignments
319 		jQuery('#pdbFile_select').html(this._createOptions(alignments));
320 		
321 		// Select an alignment
322 		if ( alignments.hasOwnProperty( selectedAlignment.slice(0,selectedAlignment.indexOf(' '))) ) {
323 			// Select current alignment if it belongs to filtered alignments
324 			jQuery('#pdbFile_select').val(selectedAlignment);
325 			
326 		} else { 
327 			// Select any alignment
328 			for ( a in alignments ){
329 				jQuery('#pdbFile_select').val(a);
330 				break;
331 			}
332 			this._onAlignmentSelectionChange();
333 		}
334 		
335 		// invoke setSelection on the parent 
336 		this.base(selection);
337 	},

339 	_createOptions: function(alignments) {
340         var pdbOptions = "";
341         for (pdb in alignments) {
342             text = pdb + " (" + alignments[pdb][1].start + ".." + alignments[pdb][1].end + ")";
343             pdbOptions += '<option value="' + text + '">' + text + '</option>';
344         }
345         Biojs.console.log("_createOptions: " + pdbOptions);
346         return pdbOptions;
347     },

349 	_filterAligmentsBySelection: function(selection) {
350         var alignments = undefined;
351         if (selection instanceof Array) {
352             alignments = {};
353             for (al in this._alignments) {
354                 var uniprot = this._alignments[al][1];
355                 for (i in selection) {
356                     if (selection[i] >= uniprot.start && selection[i] <= uniprot.end) {
357                         alignments[this._alignments[al][0].intObjectId] = this._alignments[al];
358                         break;
359                     }
360                 }
361             }
362         } else if (selection instanceof Object) {
363             alignments = {};
364             var i = 0;
365             for (al in this._alignments) {
366                 var uniprot = this._alignments[al][1];
367                 if ((selection.start >= uniprot.start && selection.start <= uniprot.end) || (selection.end >= uniprot.start && selection.end <= uniprot.end) || (selection.start < uniprot.start && selection.end > uniprot.end)) {
368                     alignments[this._alignments[al][0].intObjectId] = this._alignments[al];
369                     i++;
370                 }
371             }
372         } else {
373             alignments = this._alignments;
374         }
375         Biojs.console.log("Filtered alignments:");
376         Biojs.console.log(alignments);
377         return alignments;
378     },

380 	/**
381     * Selection of a region using the uniprot positions.
382     *
383     * @example 
384     * // Selection of the region in the interval [100,150].
385     * instance.setSelection({start: 100, end: 150});
386     * 
387     * @example
388     * // Selection of the positions 4, 8 and 100.
389     * instance.setSelection([4,8,100]);
390     * 
391     * @param {Object|Array} selection Can be either a plain object or an array.  
392     *        If object, it must have the fields start and end; Where "start" is greater than or equal to "end".
393     *        If array, it must contain numbers representing the positions to be selected.
394     */ 
395 	setSelection: function ( s ) {
396 		
397 		var selection = Biojs.Utils.clone(s);
398 		var alignmentId = this.getCurrentAlignmentId();
399 		var proteinId = this.getCurrentProteinId();
400 		var segment = this.getAlignmentsByPdb(alignmentId)[alignmentId];
401 		var offset = 0;
402 		
403 		for ( i in segment ) {
404 			if ( segment[i].intObjectId == alignmentId ) {
405 				pdbSegment = segment[i];
406 			} else if ( segment[i].intObjectId == proteinId ) {
407 				uniprotSegment = segment[i];
408 			}
409 		}
410 		
411 		offset = uniprotSegment.start - pdbSegment.start;
412 		
413 		if ( selection instanceof Array ) {
414 			for ( i in selection ) {
415 				selection[i] -= offset;
416 			}
417 		} else if ( selection instanceof Object && selection.start <= selection.end ){
418 			selection.start -= offset;
419 			selection.end -= offset;
420 		}
421 		
422 		this.base(selection);
423 	},
424

427 		var alignmentId = selectedValue.substring(0, selectedValue.indexOf(' '));
428 		//var pdbId = alignmentId.substring(0, pdb.indexOf('.')).toLowerCase();
429 		return alignmentId;
430 	},
