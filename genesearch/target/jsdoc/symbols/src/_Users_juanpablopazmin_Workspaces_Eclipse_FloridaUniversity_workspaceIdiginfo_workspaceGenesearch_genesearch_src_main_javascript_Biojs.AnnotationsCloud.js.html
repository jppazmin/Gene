 33 /** @lends Biojs.AnnotationsCloud# */

 44 	},

 47 	 * Default values for the options

 49 	 */
 50 	opt: {

 52 		sparqlUrl: 'http://biotea.idiginfo.org/sparql',
 53 		//sparqlUrl: 'http://virtuoso.idiginfo.org/sparql',
 54 		proxyUrl: 'biojs/dependencies/proxy/proxy.php',
 55 		groupBy: Biojs.Annotations.GROUP_BY_TERM,
 56 		filterBy: { field:"topic", value:"uniprot|GO|CHEBI|ICD9|UMLS|fma|MSH|PO|MDDB|NDDF|NDFRT|NCBITaxon" }
 57 	},
 58   
 59 	/**
 60 	 * Array containing the supported event names
 61 	 * @name Biojs.AnnotationsCloud-eventTypes

 63 	eventTypes: [
 64 	    /**
 65 	     * @name Biojs.AnnotationsCloud#onTopicOver

 67 	     * @param {function} actionPerformed A function which receives an {@link Biojs.Event} object as argument.
 68 	     * @eventData {Object} source The component which did triggered the event.
 69 	     * @eventData {string} type The name of the event.
 70 	     * @eventData {string} tagName Name of the clicked tag.
 71 	     * @example 
 72 	     * instance.onTopicOver(
 73 	     *    function( e ) {
 74 	     *       alert("Topic : " + e.topicType + ", Identifier: " + e.topicId );
 75 	     *    }
 76 	     * ); 
 77 	     * 
 78 	     */
 79 	    "onTopicOver",
 80 	    "onTopicClicked",
 81 	    "onAnnotationClicked",
 82 	    "onAnnotationOver"
 83 	],
 84   
 85 	_initialize: function ( container ) {
 86 	  
 87 		var opt = this.opt;
 88 		
 89 		if ( opt.title !== undefined ) {
 90 			this._title = jQuery('<div>' + opt.title + '</div>').appendTo( container );
 91 		}
 92 
 93 		this._tags = jQuery('<ul class="tags"></ul>').appendTo( container );
 94 	},
 95 	
 96 	/**
 97      * Filters the current data. It does not a new request for data.
 98      * It just filters the already gotten data. 
 99 	 * 
100 	 * @param {object} filter It must have both members frequency and topics. 
101 	 *    filter.frequency must be an integer which filters by the minimum frequency.
102 	 *    filter.topics must be an array containing the name of topics to avoid.
103 	 * 
104 	 * @returns {object} Filtered data.
105 	 * 
106 	 * @example 
107 	 * instance.applyFilter({ frequency: 7, topics: ['uniprot','chebi'] });
108 	 * 
109 	 */
110 	applyFilter: function ( filter ) {
111 		this._annotations.applyFilter(filter);
112 		this.repaint();
113 	},
114 	
115 	/**
116      * Request the annotations for an URI resource and fills up the cloud with them.
117 	 * 
118 	 * @param {string} resourceURI Universal resource identifier 
119 	 *
120 	 * @example 
121 	 * instance.requestAnnotations("http://biotea.idiginfo.org/pubmedOpenAccess/rdf/PMC3018821");
122 	 * 
123 	 * @example 
124 	 * instance.requestAnnotations(["http://biotea.idiginfo.org/pubmedOpenAccess/rdf/PMC3018821","<http://biotea.idiginfo.org/pubmedOpenAccess/rdf/PMC3215666>"]);
125 	 * 
126 	 */
127 	requestAnnotations: function ( resourceURI, fnCallback ) {
128 
129 		//TODO: show loading image
130 		var self = this;
131 
132 	    this.removeAll();
133         this._container.addClass('loading');
134         
135         this._annotations.requestAnnotations(resourceURI, function ( data ) {
136 			
137 			self.repaint();
138 			
139 		    if (fnCallback) {
140 		    	fnCallback.call();
141 		    }
142 			
143 		});
144 
145 	},
146

148 		var self = this;
149

151         this._container.addClass('loading');
152         this._annotations = a;
153         this.repaint();
154 	},
155

160 	repaint: function () {
161 		
162 		var self = this;
163 
164 		this.removeAll();
165 		this._paintAnnotations();
166 		this._container.removeClass('loading');
167 		
168 		this.onTagOver( function(e) {
169 			var annotation = self._getAnnotation(e.tagId);
170 			// raising event 
171 			self.raiseEvent(Biojs.AnnotationsCloud.EVT_ON_ANNOTATION_OVER, {
172 				name: annotation.name,
173 				freq: annotation.freq,	
174 				comments: self._getComments(annotation)
175 			});
176 		}); 
177 	},

179     _getAnnotation: function( key ) {
180         return this._annotations.getItem(key);
181     },

183 	_paintAnnotations: function () {
184 
185 	    var self = this;
186 	    var i = this._annotations.iterator();
187 	    var a;
188 
189 	    while ( i.hasNext() ) {
190 
191 	    	a = i.next();
192 	    	
193 	    	Biojs.console.log("ADDING");
194 	    	Biojs.console.log( a );
195 
196             this.addTag({ id: a.id, name: a.name, freq: a.freq },
197                 // Function to be triggered whenever a tag is clicked
198                 function ( tagId, tagName, tagValueNode ) {
199 
200             		Biojs.console.log("Searching for tag " + tagId);
201             	
202                     var annot = self._getAnnotation(tagId);
203                     
204                     Biojs.console.log(annot);
205                     
206                     // raising event
207                     self.raiseEvent(Biojs.AnnotationsCloud.EVT_ON_ANNOTATION_CLICKED, {
208                         name: annot.name, 
209                         freq: annot.freq,
210                         comments: self._getComments(annot)
211                     });
212 
213                     var topicList = jQuery('<span class="topics"></span>');
214                     var e, anchor; 
215 
216                     var items = ( self.opt.groupBy == Biojs.Annotations.GROUP_BY_TERM) ? annot.topics : annot.terms;
217                     
218                     for ( key in items ) {
219 
220                         e = items[key].map;
221                         anchor = jQuery('<a class="topic ' + key + '">' + key + '</a>')
222                         
223                         if ( !Biojs.Utils.isEmpty(e) ) {
224                         	
225                         	anchor.attr({
226                                 href: e.baseUrl,
227                                 target: '_blank',
228                                 topicType: e.shortName
229                             })
230                             .css('color',e.color)
231                             
232                         } else {
233                         	
234                         	anchor.attr({
235                                 topicType: key
236                             })
237                         }
238 
239                         anchor.appendTo(topicList);
240 
241                     }
242                     
243                     // Draw a tooltip containing the topics of the topicType selected
244                     // Example: the identifiers of uniprot
245                     var simpleTip = new Biojs.Tooltip({
246                         targetSelector: topicList.children(),
247                         cbRender: function( element ) {
248                             return self._paintTopics( tagId, jQuery(element).attr('topicType') );
249                         },
250                         arrowType: Biojs.Tooltip.ARROW_TOP_MIDDLE
251                     });
252 
253                     return topicList;
254                 }
255             );
256         }

259 	_paintTopics: function ( tagId, topicType ) {
260 
261         var self = this;
262         var annotation = self._getAnnotation(tagId);
263         var pattern=/(#|_|:|\/)([A-Za-z0-9])*$/g;
264 
265         
266         Biojs.console.log("tagID " + tagId + ", topicType " + topicType);
267         
268         // raising event
269         self.raiseEvent(Biojs.AnnotationsCloud.EVT_ON_ANNOTATION_CLICKED, {
270             name: annotation.name,
271             freq: annotation.freq,
272             comments: self._getComments(annotation)
273         });
274 
275         var topicList = jQuery('<span class="topics"></span>');
276         var topic, identifier;
277         
278         var topicsByType = ( this.opt.groupBy == Biojs.Annotations.GROUP_BY_TERM) ? annotation.topics[topicType] : annotation.terms[topicType];
279         
280         for ( i in topicsByType.entities ) {
281 
282         	//topic = topicsByType.map;
283         	topic = ( this.opt.groupBy == Biojs.Annotations.GROUP_BY_TERM) ?  topicsByType.map : annotation.map;
284         	
285         	identifier = ( topicsByType.entities[i].match(pattern)[0] ).substr(1);
286 
287             jQuery('<a class="topic ' + topic.shortName + '">' + topic.prefix + identifier + '</a>')
288                 .attr({
289                     href: topic.baseUrl + identifier,
290                     target: '_blank',
291                     topicType: topic.shortName,
292                     topicId: identifier
293                 })
294                 .css('color',topic.color)
295                 .mouseover( function(e){
296                     var target = jQuery(e.target);
297                     self.raiseEvent(Biojs.AnnotationsCloud.EVT_ON_TOPIC_OVER, {
298                         href: target.attr("href"),
299                         topicType: target.attr("topicType"),
300                         topicId: target.attr("topicId")
301                     });
302                 })
303                 .click(function(e){
304 
305                     // Do not open the href
306                     e.preventDefault();
307 
308                     var target = jQuery(e.target);
309                     self.raiseEvent(Biojs.AnnotationsCloud.EVT_ON_TOPIC_CLICKED, {
310                         href: target.attr("href"),
311                         topicType: target.attr("topicType"),
312                         topicId: target.attr("topicId")
313                     });
314                 })
315                 .appendTo(topicList);
316 
317         }
318 
319         return topicList;
320     },
321 
322     _getComments: function( annotation ) {
323     	
324     	var compareTo = 30;
325     	var comments = [];
326     	var comment;
327     	var last="";
328     	var t = annotation.name;
329     	
330     	for ( var i in annotation.comments ) {
331     		
332     		if ( ! ( annotation.comments[i][0].slice(0, compareTo) == last ) ) {
333     			comment = annotation.comments[i][0].replace( new RegExp( t.toLowerCase(), 'gi' ), "<span class='highlight'>" + t + "</span>" );
334     			comments.push( [ comment ] );
335     		}
336     		last = annotation.comments[i][0].slice(0, compareTo);
337     	}
338     	
339     	return comments;
340     	
341     }
