 80 	constructor: function (options) {
 81 
 82 		var self = this;
 83 		
 84 		var width = this.opt.width;
 85 		var height = this.opt.height;
 86 		
 87 		this._appletId = "jmolApplet" + self.getId();
 88 		
 89 		this._container = jQuery('#' + this.opt.target).addClass('Protein3D');
 90 		
 91 		// Container Width 
 92 		if ( width == undefined ) {
 93 			width = this._container.css('width');
 94 			
 95 		} else {
 96 			this._container.width( width );
 97 		}
 98 		
 99 		// Container Height
100 		if ( height == undefined ) {
101 			height = this._container.css('height');
102 			
103 		} else {
104 			this._container.height( height );
105 		}
106 
107 		this.opt.backgroundColor = ( this.opt.backgroundColor !== undefined )? this.opt.backgroundColor : this._container.css('background-color');
108 
109 		this._container.html('');
110 		
111 		this._controlsContainer = jQuery('<div id="controlSection" class="Protein3D_tab"></div>')
112 			.appendTo( this._container );
113 		
114 		this._appletContainer = jQuery('<div id="div'+self._appletId+'" class="Protein3D_applet"></div>')
115 			.appendTo( this._container );
116 		
117 		this._loadingImage = jQuery('<div class="Protein3D_loadingImage" />')
118 			.appendTo( this._container )
119 			.hide();
120 
121 		if (this.opt.loadingStatusImage !== undefined ) {
122 			this._loadingImage.css( 'background-image', 'url(' + this.opt.loadingStatusImage + ')' );
123 		}
124 		
125 		this._initializeJmolApplet();
126 		
127 		Biojs.console.log("ending Biojs.Protein3D constructor");
128    },

224 	   jmolSetAppletColor( self.opt.backgroundColor );

359 	setPdb: function( pdb ){
360 		Biojs.console.log("LOADING pdb content");
361 		//Biojs.console.log(pdb);
362 		
363 		var self = this;
364 		var surfaceCmd = this._getDisplaySurface( this.opt.surface );
365 		var styleCmd = this._getDisplayStyle( this.opt.style );
366 		var colorSchemeCmd = this._getDisplayColor( this.opt.colorScheme );

381 		Biojs.console.log("setPdb() ending");

403    /**
404     * Filters the currently provided PDB files: Only PDB files containing a part of the requested region
405     * are selectable. The specified region is highlighted in the displayed PDB file.
406     *
407     * @example 
408     * // Selection of the region in the interval [100,150].
409     * instance.setSelection({start: 100, end: 150});
410     * 
411     * @example
412     * // Selection of the positions 4, 8 and 100.
413     * instance.setSelection([4,8,100]);
414     * 
415     * @param {Object|Array} selection Can be either a plain object or an array.  
416     *        If object, it must have the fields start and end; Where "start" is greater than or equal to "end".
417     *        If array, it must contain numbers representing the positions to be selected.
418     */ 
419 	setSelection: function(selection){
420 		if ( selection instanceof Array ) {
421 			this._selection = Biojs.Utils.clone(selection);
422 			this._drawSelection();
423 			this.raiseEvent( Biojs.Protein3D.EVT_ON_SELECTION, {
424 				selectionType: "positions",
425 				selection: Biojs.Utils.clone(selection)
426 			});
427 		}
428 		else if ( selection instanceof Object && selection.start <= selection.end ){
429 			this._selection = Biojs.Utils.clone(selection);
430 			this._drawSelection();
431 			this.raiseEvent( Biojs.Protein3D.EVT_ON_SELECTION, {
432 				selectionType: "region",
433 				selection: Biojs.Utils.clone(selection)
434 			});
435 		} else {
436 			Biojs.console.log("selection not valid");
437 		}
438 	},
439 
440 	// highlight given region
441 	_getSelectionScript: function(selection){
442 		var scr = "";
443 		var selectionText = "";
444 		
445 		if ( selection ){
446 			scr = 'select all; color translucent 1;';
447 			selectionText = (selection instanceof Array)? selection.join(",") : selection.start + "-" + selection.end;
448 			
449 	        if ( !this._display.halos ) {
450 	            scr += 'select not ' + selectionText + '; color translucent 0.8; selectionHalos off;';
451 	        } else {
452 	            scr += 'select ' + selectionText + '; selectionHalos on;';
453 	        }
454 		} else {
455 			scr = "select none"; 
456 		}
457 		
458 		Biojs.console.log("Selection script: " + scr);
459 		
460 		return scr;		
461 	},

465 			result = jmolScriptWait( this._getSelectionScript(this._selection), this.getId() );
466 			Biojs.console.log("Selection done, result: "+result); 
467 		} 
468 	},
