174     /**
175     * Sets the protein accession which changes everything to reflect the new protein information
176     *
177     * @param {string} accession The new protein identifier
178     *
179     * @example
180     * instance.setProtein("P07148");
181     *
182     * @example
183     * instance.setProtein("P12345");
184     */
185     setProtein: function( accession ) {
186 
187         this.opt.accession = accession;
188 
189         this._requestAligments(accession, function(alignments){
190 
191             this._alignments = alignments;
192 
193             this._container.removeClass('loading');
194 
195             if (!Biojs.Utils.isEmpty(alignments)) {
196                // Alignments found
197                this._setHeader( this.opt.accession, this._createPDBSelect(alignments) );
198 
199             } else {
200                // none alignment found
201                this._setHeader( this.opt.accession );
202             }
203         });
204 
205         this._showFeatures(accession);
206     },
207 
208     _requestAligments: function( accession, action ){
209         var self = this;
210 
211         this._clearAll();
212         this._container.addClass('loading');
213 
214         var xhr = jQuery.ajax({
215             url: self.opt.mappingUrl + accession,
216             data: {biojsmapping:'1', varname:'pdbmappings'},
217             dataType: "script",
218             crossDomain: true,
219             timeout: 5000,
220             success: function(){
221                 Biojs.console.log("SUCCESS: data received");
222                 self._alignments = pdbmappings;
223                 action.call(self,pdbmappings);
224             },
225             async: true,
226             error: function(qXHR, textStatus, errorThrown){
227                 Biojs.console.log("ERROR: " + textStatus );
228                 self.raiseEvent('onRequestError', {message: textStatus});
229                 action.call(self,{});
230             }
231         });
232     },

234     _createPDBSelect: function(alignments) {
235         var pdbOptions = "";
236         for (pdb in alignments) {
237             text = pdb + " (" + alignments[pdb][1].start + ".." + alignments[pdb][1].end + ")";
238             pdbOptions += '<option value="' + text + '">' + text + '</option>';
239         }
240         Biojs.console.log("_createOptions: " + pdbOptions);
241         return pdbOptions;
242     },

244     _setHeader: function( accession, mappingsSelector ) {
245         var self = this;
246 
247         if ( mappingsSelector !== undefined ) {
248             this._header.html("Accession: "+ accession + ", PDB Alignments: <select>"+ mappingsSelector + "</select>" );
249             this._onAlignmentSelectionChange();
250             this._header.children('select').change( function(){ self._onAlignmentSelectionChange() });
251         } else {
252             this._header.html("Accession: "+ accession + ", No PDB alignments found." );
253         }
254         
255         this._adjustContentSize();
256 
257     },
258 
259     _onAlignmentSelectionChange: function(){
260 
261         var alignment = this.getCurrentAlignment();
262 
263         if ( !Biojs.Utils.isEmpty(alignment) ) {
264 
265             this.raiseEvent('onPdbSelected', {
266                                 "pdbId": alignment.pdbId,
267                                 "alignmentId": alignment.alignmentId,
268                                 "start": alignment.start,
269                                 "end": alignment.end
270                             });
271 
272             this._showPDBPrints(alignment.pdbId);
273             this._showImage(alignment.pdbId);
274             this._showFeatures(this.opt.accession);
275 
276         } else {
277             Biojs.console.log("No structural information available for " + this.opt.accession );
278         }
279     },

281     _showPDBPrints: function(pdbId) {
282 
283         var self = this;
284         this._toolbar.children().remove();
285         this._toolbar.attr('id', 'PDBLogos_toolbar_' + this.getId() );
286 
287         var myprints = new Biojs.PDBLogos({
288             target: this._toolbar.attr('id'),
289             identifier: pdbId
290         });
291 
292         myprints.onClick( function(e) {
293         	if ( e.category == "PDBStructure" ) {
294         		self._showProtein3D();
295         	} else {
296         		window.open( e.printsURL );
297         	}
298         });
299 
300         this._adjustContentSize();
301     },

303     /**
304     * Get the available alignments for the current protein filtered by selection (PDB files containing a part of the requested region).
305     *
306     * @example
307     * // Get the alignments matching with bases 4, 8 and 100.
308     * alert ( instance.getAlignmentsByPdb("2F73") );
309     *
310     * @param {string} pdb identifier.
311     * @returns {Object} .
312     *
313     */
314     getAlignmentsByPdb: function ( pdbId ) {
315 
316         var alignments = {};
317 
318         for (al in this._alignments) {
319             if (  this._alignments[al][0].intObjectId.indexOf( pdbId ) != -1 ) {
320                 alignments[this._alignments[al][0].intObjectId] = this._alignments[al];
321             }
322         }
323 
324         Biojs.console.log("Alignments for pdb " + pdbId );
325         Biojs.console.log(alignments);
326 
327         return alignments;
328     },
329 
330     getCurrentAlignment: function(){
331 
332         var result = {};
333         var pdbSelectedValue = this._header.find('select').val();
334 
335         if ( pdbSelectedValue != undefined ) {
336 
337             result.accession = this.opt.accession;
338             result.alignmentId = pdbSelectedValue.substring(0, pdbSelectedValue.indexOf(' '));
339             result.pdbId = pdbSelectedValue.substring(0, pdbSelectedValue.indexOf('.')).toLowerCase();
340 
341 
342             var alignment = this.getAlignmentsByPdb(result.alignmentId);
343 
344             result.start = alignment[result.alignmentId][1].start;
345             result.end = alignment[result.alignmentId][1].end;
346             result.offset = 0;
347 
348             if ( alignment.hasOwnProperty( result.alignmentId ) ) {
349                 result.offset = result.start - alignment[result.alignmentId][0].start;
350             }
351 
352             result.alignment = alignment;
353         }
354 
355         Biojs.console.log("Current alignment");
356         Biojs.console.log(result);
357 
358         return result;
359     },

361     _showProtein3D: function( ){
362 
363         var self = this;
364         var alignment = this.getCurrentAlignment();
365         var height = this._container.height() - this._header.height() - this._toolbar.height();
366 
367         self._content.children().remove();
368         self._beta.children().remove();
369 
370         self._beta.attr('id','biojs_protein3D_' + self._content.attr('id') );
371 
372         // Calculate character width
373         var test = jQuery("<span>X</span>").appendTo(self._content);
374         var columns = Math.round( self._content.width() * (1/test.width()) );
375         test.remove();
376 
377         var sequence = new Biojs.Sequence({
378             target : self._content.attr('id'),
379             format : 'FASTA',
380             id : alignment.accession,
381             formatSelectorVisible: false,
382             columns: { size: columns, spacedEach: 10 }
383         });
384 
385         var proteinStructure = new Biojs.Protein3DWS({
386             id: alignment.pdbId,
387             target: self._beta.attr('id'),
388             width: Math.round( self._beta.width() ),
389             height: self._container.height(),
390             proxyUrl: self.opt.proxyUrl,
391             jmolFolder: self.opt.jmolFolder
392         });
393 
394         sequence.onSelectionChange( function (e) {
395             // [e.start, e.end] from the event data corresponds to the protein sequence positions
396             // which need to be aligned with PDB structure
397             var selection = {start: e.start, end: e.end};
398 
399             selection.start -= alignment.offset;
400             selection.end -= alignment.offset;
401 
402             proteinStructure.setSelection( selection );
403         });
404 
405     },

432     _showFeatures: function( accession ) {
433 
434         var self = this;
435 
436         this._content.children().remove();
437         this._adjustContentSize();
438 
439         this._requestFeatures( accession, function(xml){
440 
441             Biojs.console.log("Features received " + xml );
442 
443             var xmlDoc = "";
444 
445             try {
446                 xmlDoc = jQuery.parseXML( xml );
447             } catch (e) {
448                 Biojs.console.log("XML parser error" + e);
449             }
450 
451             var features = jQuery('<ul class="features"></ul>').appendTo(self._content);
452             var label="", value="";
453 
454             jQuery(xmlDoc).find('FEATURE')
455                 .each( function(){
456                     //features.push( self._decodeFeature(this) );
457                     label = '<span class="label">' + jQuery(this).attr('label') + '</span>';
458                     value = '<span class="value">' + jQuery(this).children('NOTE:first').text() + '</span>';
459 
460                     features.append('<li>' + label + value + '</li>');
461                 });
462         });
463     },
464 
465     /*
466      * Function: Biojs.ProteinPortafolio._requestFeatures
467      * Purpose:  Request data to the server
468      * Returns:  -
469      * Inputs:   opt -> {Object} options object.
470      */
471     _requestFeatures: function( accession, action ){
472 
473         var self = this;
474 
475         Biojs.console.log("Requesting features for '" + accession + "'" );
476 
477         var httpRequest = {
478             url: self.opt.proxyUrl,
479             data: [{ name: "url", value: self.opt.featuresUrl + "?segment=" + accession }],
480             method: "GET",
481             dataType: "text",
482             success: action,
483             error: function(e) {
484                 Biojs.console.log("Error requesting features for " + accession );
485             }
486         };
487 
488         jQuery.ajax(httpRequest);
489 
490     },
491     
492     _adjustContentSize: function() {
493 
494     	if ( this._beta.children().length > 0 ) {
495     		this._alpha.addClass('alpha');
496     		this._beta.addClass('beta');
497     		this._content.height(this._container.height() - this._header.height() - this._toolbar.height());
498     		
499     	} else {
500     		this._alpha.removeClass('alpha');
501     		this._beta.removeClass('beta');
502     		this._content.height('auto');
503     	}
504     }
