1 /** 
  2  * Table to show binary molecular interactions.  
  3  * 
  4  * @class
  5  * @extends Biojs.Table
  6  * 
  7  * @option {string[][]|Object} dataSet 
  8  *    Either 2D string array containing the whole data to be displayed or a plain object defining the data source. 
  9  *    
 10  *    <pre class="brush: js" title="Syntax of the plain object:">
 11  *    {
 12  * 		url: <url>,
 13  * 		proxyUrl: <proxy>,
 14  * 		paramsMap: { "iDisplayStart": <newName1>, "iDisplayLength": <newName2>, ... },
 15  * 		filter: <flag>,
 16  * 		version: <MITAB_VERSION>,
 17  *      totalRecords:<number>
 18  *    }
 19  *    </pre>
 20  *    where:
 21  *      <ul>
 22  *        <li><b>url</b> is string containing the url data source.</li>
 23  *        <li><b>paramsMap</b> Customize the name of the following params which will be passed to server:
 24  *    			<ul>
 25  *    				<li>iDisplayStart: index of the first expected record for paging.</li>
 26  *    				<li>iDisplayLength: size of the fetched data (page size).</li>
 27  *    				<li>sSearch: filter string entered by the user in case of filter is enabled.</li>
 28  *    			</ul>
 29  *        </li>
 30  *        <li><b>filter</b> is a boolean to show/hide the search box on the table top. 
 31  *        The entered string will be passed to the server by means of 'sSearch' parameter.
 32  *        Personalize the parameter name using paramsMap.</li>
 33  *        <li><b>proxyUrl</b> optional string containing the url of the proxy in case of needing access to server through a proxy.</li>
 34  *        <li><b>version</b> either string or integer containing the version of MITAB format as follows:
 35  *        	<ul>
 36  *    			<li>"MITAB_VERSION_2_5" or 15: to set <a href="http://code.google.com/p/psicquic/wiki/MITAB25Format">MITAB format v2.5</a></li>
 37  *    			<li>"MITAB_VERSION_2_5_EXT" or 31: to set MITAB format v2.5 extended</li>
 38  *    			<li>"MITAB_VERSION_2_6" or 36: to set <a href="http://code.google.com/p/psicquic/wiki/MITAB26Format">MITAB format v2.6</a></li>
 39  *    			<li>"MITAB_VERSION_2_7" or 42: to set <a href="http://code.google.com/p/psicquic/wiki/MITAB27Format">MITAB format v2.7</a></li>
 40  *    		</ul>
 41  *        </li>
 42  *        <li><b>totalRecords</b> integer containing the number of total records. MITAB format does not contains this value on the response.
 43  *        You must do a request to ask for it.</li>
 44  *      </ul>
 45  *      
 46  *    <pre class="brush: js" title="Example of plain object: ">
 47  *    {
 48  * 		psiquicUrl: 'http://www.ebi.ac.uk/Tools/webservices/psicquic/intact/webservices/current/search/query',
 49  * 		proxyUrl: '../biojs/dependencies/proxy/proxy.php',
 50  *      version: "MITAB_VERSION_2_5_EXT",
 51  *      query: "species:human",
 52  *      filter: false
 53  *    }
 54  *    </pre>  
 55  *  
 56  *    <pre class="brush: js" title="Example of 2D array of data: ">
 57  *    ...soon
 58  *    </pre>
 59  * 
 60  * @example 
 61  * var myTable = new Biojs.InteractionsTable({
 62  *  	target: "YourOwnDivId",
 63  *  	dataSet: {
 64  *  		psicquicUrl: 'http://www.ebi.ac.uk/Tools/webservices/psicquic/intact/webservices/current/search/query',
 65  *  		proxyUrl: '../biojs/dependencies/proxy/proxy.php',

141 	 * Function: Biojs.InteractionTable._requestQueryCount
142 	 * Purpose:  Request total rows counting with the provided query for pagination purposes. 
143 	 * 			 Since the PSIQUIC service does not provides the total row count with retrieved data,
144 	 * 			 it's necessary to do another request in order to get it.
145 	 * Inputs:   dataSet -> {Object} Settings of the data set. 
146 	 * 			 action -> {function} Callback function having the dataSet as argument.
147 	 */
148 	_requestQueryCount: function ( dataSet, action ) {
149 		var self = this;
150 		
151 		dataSet.url = dataSet.psicquicUrl + '/' + dataSet.query;
152 		dataSet.paramsMap = { "iDisplayStart": "firstResult", "iDisplayLength": "maxResults" };
153 		
154 		jQuery.ajax({ 
155 			url: dataSet.proxyUrl,
156 			dataType: "text",
157 			data: [{ name: "url", value: dataSet.url + '?format=count' }],
158 			success: function ( data ) {
159 				dataSet.totalRecords = parseInt(data) | 0;
160 				action.call( self, dataSet );
161 			}
162 		});
163 	},

165     /**
166      * Rebuild the table
167      * 
168      * @example 
169      * myTable.setDataSource({
170      * 		psicquicUrl: 'http://www.ebi.ac.uk/Tools/webservices/psicquic/intact/webservices/current/search/query',
171      *  	proxyUrl: '../biojs/dependencies/proxy/proxy.php',
172      *  	query: "pubid:(10837477 OR 12029088)",
173      *      version: "MITAB_VERSION_2_5_EXT",
174      *      filter: false
175 	 * });
176      * 
177      */
178     setDataSource: function ( dataSet ) {
179     	
180     	if ( dataSet instanceof Array ) {
181     		// Local data
182     		// invoke parent's setDataSource
183     		this.base( dataSet );
184     		
185     	} else {
186     		// Remote data 
187         	if ( dataSet.hasOwnProperty("totalRecords") && dataSet.hasOwnProperty("url") ) {
188         		// Already have the number of records for pagination 
189         		// then, invoke parent's setDataSource
190         		this.base( dataSet );
191         		
192         	} else {
193         		// Set the new query
194         		this.opt.dataSet = {};
195         		this.opt.dataSet.psicquicUrl = dataSet.psicquicUrl;
196         		this.opt.dataSet.proxyUrl = dataSet.proxyUrl;
197         		this.opt.dataSet.version = dataSet.version;
198         		this.opt.dataSet.filter = dataSet.filter;
199         		this.opt.dataSet.query = escape(dataSet.query);
200         		this.opt.dataSet.url = dataSet.psicquicUrl + '/' + dataSet.query;
201         		this.opt.dataSet.paramsMap = { "iDisplayStart": "firstResult", "iDisplayLength": "maxResults" };
202         		// Request query counting rows and apply setDataSource then
203         		this._requestQueryCount( this.opt.dataSet, this.base );
204         	}
205     	}

496 	MITabTxtDecoder: function (str) {
497 		this.buffer = ( str != undefined )? str : "" ;
498 		this.offset = 0;
499 		/**
500 		 * @ignore
501 		 */
502 		this.nextToken = function (delimiter) {
503 			var token = "";
504 			var index;
505 			if ( this.offset < this.buffer.length ) {
506 				start = this.offset;
507 				index = this.buffer.indexOf(delimiter, start);
508 				
509 				if ( index != -1 ) {
510 					this.offset = index;
511 					token = this.buffer.substring(start, index);
512 					this.offset += delimiter.length;
513 				} else {
514 					this.offset = this.buffer.length;
515 					token = this.buffer.substring(start);
516 				}
517 			} else {
518 				// no more tokens
519 			}			
520 			return token;
521 		}
522 		/**
523 		 * @ignore
524 		 */
525 		this.hasNext = function () {
526 			return this.offset < this.buffer.length;
527 		}
528 		/**
529 		 * @ignore
530 		 */
531 		this.decodeXRef = function () {
532 			
533 			var token = "";
534 			var xref = { key:"", value:"", desc:"" };
535 			
536 			token = this.nextToken(":");
537 				
538 			if ( token.length > 1 ) {
539 					
540 				xref.key = token;
541 				xref.value = this.nextToken("|").replace(/\"/gi,'');
542 				
543 				descIdx = xref.value.indexOf("(");
544 				
545 				if ( descIdx != -1 ) {
546 					xref.desc = xref.value.substring( descIdx +1, xref.value.length -1);
547 					xref.value = xref.value.substring( 0, descIdx);
548 				}
549 				
550 				Biojs.console.log("XREF<"+ xref.key +">, VALUE<"+ xref.value +">, DESC<"+ xref.desc +">" );
551 			}
552 			
553 			return xref;
554 		}
555 		/**
556 		 * @ignore
557 		 */
558 		this.decodeRow = function () {
559 			var row = [];
560 			if ( this.hasNext() ) {
561 				row = this.nextToken("\n").split("\t");
562 			}
563 			return row;
564 		}
565 		/**
566 		 * @ignore
567 		 */
568 		this.decodeData = function () {
569 			var data = [];
570 			
571 			while ( this.hasNext() ){
572 				data.push(this.decodeRow());
573 			}
574 			return data;
