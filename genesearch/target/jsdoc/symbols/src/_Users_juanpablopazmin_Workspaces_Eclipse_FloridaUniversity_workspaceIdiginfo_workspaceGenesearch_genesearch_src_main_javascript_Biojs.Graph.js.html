163 	 * 
164 	 * @param {string} [id] 
165 	 *  Optional identifier for the new link
166 	 *
167 	 * @returns {object} Reference to the link created.

170 	 * // Creates a link with label Catalase and id linkCatalase

174 	 * instance.addLink('element5','element6','Testing');

177 	addLink: function(source, target, label, id) {
178 		
179 		var idSource = source;
180 		var idTarget = target;
181 		var link;
182 		var settings;
183 		var self = this;
184 
185 		// Creates the new connection
186 
187 		if ( "number" == typeof source ) {
188 			idSource = (source).toString();
189 			
190 		} else if ( "object" == typeof source ) { 
191 			idSource = jQuery(source).attr('id');
192 		}
193 		
194 		if ( "number" == typeof target ) {
195 			idTarget = (target).toString();
196 			
197 		} else if ( "object" == typeof target ) {
198 			idTarget = jQuery(target).attr('id');
199 		}
200 
201 		settings = {
202 			source:idSource, 
203 			target:idTarget,
204 		};
205 		
206 		if ( id != undefined ) {
207 			settings.id = id;
208 		}
209 		
210 		// Label
211 		settings.overlays = [ 
212 			[ "Label", {													   					
213 	   			cssClass:"label",
214 	   			label : label, 
215 	   			id:"label",
216 	   			events:{
217 					"click":function(label, evt) {
218 						self.raiseEvent(Biojs.Graph.EVT_ON_LABEL_CLICKED, {
219 							'label': label
220 						});
221 	   				}
222 	   			}
223 			  }
224 			]
225 		];
226 		
227 				
228 		try {
229 			link = this._jsPlumb.connect(settings);
230 			
231 			var overlay = link.getOverlay("label").hide();
232 			
233 			// Add behavior for the connection 
234 			link.bind("mouseenter", function(conn) {
235 					var overlay = conn.getOverlay("label");
236 					// now you can show this Overlay:
237 					overlay.show();
238 					// trigger the event on connection over 
239 					self.raiseEvent(Biojs.Graph.EVT_ON_LINK_OVER, {
240 						'link': link,
241 						'label': conn.getOverlay("label"),
242 						'source': idSource,
243 						'target': idTarget
244 					});
245 					
246 				}).bind("mouseexit", function(conn) {
247 					var overlay = conn.getOverlay("label");
248 					// now you can hide this Overlay:
249 					overlay.hide();
250 				});
251 			
252 		} catch (e) {
253 			Biojs.console.log(e);
254 		}
255 		
256 		var nodeSource = this._nodes[idSource];
257 		var nodeTarget = this._nodes[idTarget];
258 		
259 		nodeSource.links[idTarget] = nodeTarget;
260 		nodeTarget.links[idSource] = nodeSource;
261 		
262 		nodeSource.linksLength++;
263 		nodeTarget.linksLength++;
264 		
265 		nodeSource.moving = true;
266 		nodeTarget.moving = true;
267 		
268 		this.start();
269 		
270 		return link;
271

274 	/**
275      * Removes a link from the graph.
276 	 * 
277 	 * @param {string|object} link 
278 	 *	Can be either an identifier as string or the link object itself returned by the method addLink.  
279 	 *
280 	 * @returns {boolean} True on successful deletion, false in other case. 
281 	 * 		
282 	 * @example  
283 	 * // Removes the link with linkCatalase as identifier.
284 	 * instance.removeLink('linkCatalase');
285 	 * 
286 	 */
287 	removeLink: function(link) {
288 
289 		var cnn = link;
290 		// If an identifier is provided
291 		// it will search for the connection object
292 		if ( "string" == typeof link ){
293 			
294 			var i;
295 			var conns = this._jsPlumb.getConnections();
296 			
297 			for ( var i=0; i < conns.length && ( link != conns[i].getId() ) ; i++ ) {
298 				; // breaks on link found
299 			}
300 			
301 			cnn = ( i < conns.length )? conns[i] : {};
302 		} 
303 		
304 		if ( cnn.hasOwnProperty("removeOverlays") ) {
305 			
306 			//Remove references from involved nodes 
307 			delete this._nodes[cnn.sourceId].links[cnn.targetId];
308 			delete this._nodes[cnn.targetId].links[cnn.sourceId];
309 			// Remove the labels
310 			cnn.removeOverlays();
311 			// remove all endpoints
312 			for ( var e in cnn.endpoints ) {
313 				this._jsPlumb.deleteEndpoint(cnn.endpoints[e]);
314 			}
315 			
316 			this.start();
317 			
318 			return true;
319 		}
320 		
321 		return false;
322 	},

333 	removeLinksFromNode: function(node) {
334 		Biojs.console.log("Removing links from node " + node );
335 		// node can be either an identifier or the node object itself
336 		var id = ( "string" == typeof node )? node : node.id;
337 		// get all the connection objects of the node

339 		// remove every single connection

342 		}

346 	/**
347      * Creates a new node into graph.
348 	 * 
349 	 * @param {string|object} data 
350 	 *	Data could be either an identifier or an HTML element. 
351 	 *	If it's an identifier, a DIV element with that identifier will be created as node. 
352 	 *	In case of HTML element, can be either a string containing the HTML code or an instance of HTMLElement. 
353 	 *
354 	 * @returns {object} The node instance created/added to the graph.	
355 	 *
356 	 * @example  
357 	 * // Creates a node with 'element1' as id
358 	 * instance.addNode('element4');
359 	 * 
360 	 * @example 
361 	 * // Creates a node with 'element1' as id and then it is assigned to var 'node' 
362 	 * var node = instance.addNode('element5');
363 	 * 
364 	 * @example 
365 	 * // Creates a node since the HTML string provided
366 	 * instance.addNode('<div id="element6"></div>');
367 	 * 
368 	 */
369 	addNode: function ( data ) {
370 		
371 		var node = jQuery('<div></div>');
372 		var self = this;
373 		
374 		if ( "string" == typeof data ) {
375 			
376 			if ( data.match(/<\/?.*(?=>|\s.*>)\/?.*?>/g) ) {
377 				node = jQuery( data );
378 				
379 			} else {
380 				node.attr('id', (!Biojs.Utils.isEmpty(data))? data : "node" + this._nodeIdCounter++ );
381 			}
382 
383 		} else if ( "number" == typeof data ) {
384 			node.attr('id', (data).toString() );
385 
386 		} else {
387 			// Object is expected 
388 			node = jQuery( data );
389 		}
390 		
391 		// Setting id for the node
392 		node.id = node.attr('id');
393 		if ( Biojs.Utils.isEmpty(node.id) ) {
394 			node.id = "node" + this._nodeIdCounter++;
395 			node.attr('id',node.id);
396 		}
397 		
398 		// TODO: improve positioning by Force-based algorithm
399 		// Randomly positioned at beginning
400 		node.x = this.opt.viewport.xMin + Math.floor( Math.random() * this.opt.viewport.xMax/3 );
401 		node.y = this.opt.viewport.yMin + Math.floor( Math.random() * this.opt.viewport.yMax/3 );
402 		
403 		//Initilizing values for Force-based algorithm
404 		node.velocity={x:0,y:0}; node.mass=1;
405 		node.fixed = false;
406 		node.moving = true;
407 		node.links = {};
408 		node.linksLength = 0;
409 		
410 		this._nodes[node.id] = node;
411 		//this._links[node.id] = [];
412 
413 		node.css({
414 				position:"absolute",
415 				top: node.x,
416 				left: node.y 
417 			})
418 			.addClass('node')
419 			.appendTo(this._container);
420 
421 		this._jsPlumb.draggable(node.id);
422 		
423 		// Node close button
424 		var closeButton = jQuery('<span class="close-btn-18" title="Close">Close</span>')
425 			.click(function(){
426 				self.removeNode(node);
427 			})
428 			.hide()
429 			.prependTo(node);
430 		
431 		//Update position in dragging 	
432 		node.mousedown(function() {
433 	        jQuery(window).mousemove(function() {
434 	        	node.x = node.offset().left;
435 	        	node.y = node.offset().top;
436 	        	
437 	        	// Release other nodes
438 	        	self._setPivotNode(node);
439 	        	
440 	        	for (var i in node.links) {
441 	        		node.links[i].moving = true;
442 	        	}
443 	        	
444 	        	self.start();
445 	        });
446 	    })
447 	    .mouseup(function() {
448 	        jQuery(window).unbind("mousemove");
449 	        self._releasePivotNode(node);
450 	    })
451 	    .mouseover(function() {
452 	    	closeButton.show();
453 	    })
454 	    .mouseout(function() {
455 	    	closeButton.hide();
456 	    });
457 		
458 		this.start();
459 
460 		return node;
461 	},

542      * Starts the nodes positioning animation by using a force-based algorithm. 
543      * Stops automatically when the system reaches the equilibrium, i.e. all nodes are stopped.
544      * 
545 	 * @example 
546 	 * instance.start();
547 	 * 
548 	 */
549 	start: function() {
550     	var self = this;
551         if(!this._running && this.opt.animate) {
552             this._running = true;
553             this._drawId = setInterval(function() { self._forceBasedAlgorithm(self.opt); }, 1000 / this.opt.fps);
554         }
555     },

565             clearInterval(this._drawId);
566             this._running = false;
567         }

706 	_setPivotNode: function( node ) {
707 
708 		if ( this._pivotNode == undefined ) {
709 			this._pivotNode = node;
710 			
711 		} else if ( node != this._pivotNode ) {
712 			// Release current pivot if there is any
713 			this._releasePivotNode( this._pivotNode, 0 );
714 			// Mark it as pivot
715 			this._pivotNode = node;
716 			
717 		} else {
718 			// Node is the current pivot already
719 			// just do cancel the timer if there is any
720 			this._cancelPivotTimer(node);
721 		}
722 		
723 		// Don't move this node
724 		node.fixed = true;
725 	},
726 	
727 	_releasePivotNode: function( node, time ) {
728 		// time waiting to be released
729 		var ms = (time != undefined) ? time : 1000;
730 		// wait?
731 		if ( ms > 0 ) {
732 			// On releasing the node, keep fixed until equilibrium
733 	        // Otherwise, all nodes could move together for long time
734 			this._setPivotTimer(node, ms);
735 			
736 		} else {
737 			// release it now! 
738 			this._cancelPivotTimer(node);
739 		}
740 		
741 	},
742 	
743 	_setPivotTimer: function(node, ms) {
744 		var self = this;
745 		this._cancelPivotTimer(node);
746 		node.fixed = true;
747 		node.waitingReleaseId = setInterval(function() { 
748         	if (self._equilibrium) {
749         		self._cancelPivotTimer(node);
750         	}
751         }, ms );
752 	},
753 	
754 	_cancelPivotTimer: function ( node ) {
755 		node.fixed = false;
756 		if ( node.waitingReleaseId > 0 ) {
757     		clearInterval(node.waitingReleaseId);
758     		node.waitingReleaseId = 0;
759 		}
760 	},

762     // Moves every node until whole system is under equilibrium
763     _forceBasedAlgorithm: function(o) {
764         
765     	// assume equilibrium - set to false below if a node moves
766         this._equilibrium = true;
767 
768         var repellingForce, attractingForce, node, otherNode, square, direction;
769         var numberOfNodes = this._nodes.length;
770         var numberOfStoppedNodes = 0;
771         var startingVelocityDelta = o.maximumStartingVelocity - o.minimumStartingVelocity;
772 
773         for(var i in this._nodes) {
774         	
775         	node = this._nodes[i];
776         	
777             square = this._createSquare(node, o.influenceConstant * o.normalDistance);
778             
779             for(var j in this._nodes) {
780             
781             	otherNode = this._nodes[j];
782             	
783                 // not the same node
784                 if(i != j) {
785                     // for nodes within range, add the repelling velocity
786                     if(this._withinSquare(otherNode, square)) {
787                     	repellingForce = this._repelling(node, otherNode);
788                     	node.velocity.x += repellingForce.x;
789                         node.velocity.y += repellingForce.y;
790                     }
791                 }
792                 // the first iteration we count the number of stopped nodes
793                 if(j == 0 && (!otherNode.moving || otherNode.fixed )) {
794                     numberOfStoppedNodes++;
795                 }
796             }
797 
798             // for all links, add the attracting velocity
799             for( var k in node.links ) {
800                 //if(link.visible) {
801 	            	attractingForce = this._attracting(node, node.links[k]);
802 	            	node.velocity.x += attractingForce.x/node.linksLength;
803 	                node.velocity.y += attractingForce.y/node.linksLength;
804                 //}
805             }
806             
807             // apply damping
808             node.velocity.x *= o.damping;
809             node.velocity.y *= o.damping;
810             node.velocityLength = this._absolute(node.velocity.x) + this._absolute(node.velocity.y);
811             
812             // check to see if the node has stopped moving
813             if(node.velocityLength <= o.stoppingVelocity){
814             	node.moving = false;
815             }
816                 
817             // the more nodes that have stopped, the larger the starting velocity must be
818             if( node.velocityLength >= o.minimumStartingVelocity + startingVelocityDelta * numberOfStoppedNodes / numberOfNodes ) {
819             	node.moving = true;
820             }
821                 
822             // Check the node is inside viewport
823             // otherwise change velocity's direction
824             this._setNodeDirection(node, this.opt.viewport);
825             
826             // if the node is moving and not fixed then calculate the new position
827             if(node.moving && !node.fixed ) {
828                 this._equilibrium = false;
829                 node.x += node.velocity.x;
830                 node.y += node.velocity.y;
831             }
832 
833             node.offset({ left: node.x, top: node.y });
834             this._jsPlumb.repaint(node.id);
835             
836         }
837         
838         //this._jsPlumb.repaintEverything();
839         
840         if (this._equilibrium) {
841         	this.stop();
842         }
843         
844         return this._equilibrium;
845         
846     },

848 	// create the square of influence
849     _createSquare: function(node, offset) {
850         return {
851             xMin: node.x - offset,
852             xMax: node.x + offset + node.width(),
853             yMin: node.y - offset,
854             yMax: node.y + offset + node.height()
855         };
856     },
857     
858     // is the node inside of the square of influence?
859     _withinSquare: function(otherNode, square) {
860         return  otherNode.x >= square.xMin &&
861                 otherNode.x <= square.xMax &&
862                 otherNode.y >= square.yMin &&
863                 otherNode.y <= square.yMax;
864     },
865     
866     _setNodeDirection: function(node, square) {
867     	
868     	if ( ( (node.x <= square.xMin) && (node.velocity.x < 0) ) ||
869     	     ( ( ( node.x + node.width() ) >= square.xMax) && (node.velocity.x > 0) ) ) {
870     		node.velocity.x *= -1;
871 
872     	} 
873     	
874     	if ( ( (node.y <= square.yMin) && (node.velocity.y < 0) ) ||
875     	     ( ( ( node.y + node.height() ) >= square.yMax) && (node.velocity.y > 0) ) ) {
876     		node.velocity.y *= -1;
877     		
878     	}
879 
880     },

882     _absolute: function(value) {
883         return value < 0 ? -value : value;
884     },
885 
886     _square: function(value) {
887         return value * value;
888     },
889     
890     _manhattanDistance: function(node, otherNode) {
891         var d = this._absolute(otherNode.x - node.x) + this._absolute(otherNode.y - node.y);
892         if(d > 0) return d;
893         // move one node a pixel to the right if both nodes overlap
894         node.x +=1;
895         return 1;
896     },
897     
898     _repelling: function(node, otherNode) {
899     	var repelling = this.opt.repellingFactor / this._square(this._manhattanDistance(node, otherNode));
900         return { x: repelling * (node.x - otherNode.x), 
901         	     y: repelling * (node.y - otherNode.y) };
902     },
903     
904     _attracting: function (node, otherNode) {
905     	var attraction = this.opt.attractionConstant * this._manhattanDistance(node, otherNode);
906     	return { x: (otherNode.x - node.x) * attraction, 
907     			 y: (otherNode.y - node.y) * attraction };        
908     }
