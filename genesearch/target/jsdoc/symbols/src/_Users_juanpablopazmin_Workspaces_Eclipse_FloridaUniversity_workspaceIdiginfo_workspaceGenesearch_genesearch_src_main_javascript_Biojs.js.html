 11 /**
 12  * @class
 13  * @param {string} eventType The name of the event.
 14  */
 15 Biojs.EventHandler = function(eventType) {
 16 	/**
 17      * The name of the event.
 18      * @type {string}
 19      */
 20 	this.eventType = eventType;
 21 	/**
 22      * Array of the registered listeners.
 23      * @type {function[]}
 24      */
 25 	this.listeners = [];
 26 	/**
 27      * Register action listeners for the event.
 28      * @param {function} actionPerformed The action listener to be registered.
 29      */
 30 	this.addListener = function ( actionPerformed ) {
 31 		if ( (typeof actionPerformed) == "function" ) {
 32 			this.listeners.push(actionPerformed);
 33 		}
 34 	};
 35 	/**
 36      * Executes all listener actions registered in the listeners field.
 37      * @param {Object} eventObject The event' object to be passed as argument to the listeners.
 38      */
 39 	this.triggerEvent = function( eventObject ) {
 40 		for ( var i in this.listeners ) {
 41 			this.listeners[i](eventObject);
 42 		}
 43 	}
 44 };

 82 	  var newObj = (obj instanceof Array) ? [] : {};
 83 	  for (i in obj) {
 84 	    if (obj[i] && typeof obj[i] == "object") {
 85 	    	newObj[i] = Biojs.Utils.clone(obj[i]);

 87 	    	newObj[i] = obj[i];

 99 		if (o instanceof Array || ("string" == typeof o) ) {

180 								
181 							} else if (msg instanceof String || typeof msg === "string") {
182 								message = msg;
183 								
184 							} else {

186 									message += '[' + i + ']=' + msg[i] + ' ';
187 							    }
188 							}
189 							
190 							textNode = Biojs.console.domDocument.createTextNode(message);
191 							line = Biojs.console.domDocument.createElement('pre');
192 							line.appendChild(textNode);
193 							Biojs.console.domDivNode.appendChild(line);
194 						};
195 						
196 					} else {
197 						// Game over! do not write logs, but let's tell to user by means an alert (sorry!)
198 						alert("Please activate the pop-up window in this page " +
199 								"in order to enable the BioJS console");
200 					}	
201 				}
202 			}
203 		},

209 /**
210  * Extend this class <tt>Biojs</tt> in order to create a new component.
211  * @param {object} instance The subclass.
212  * @param {object} interface (optional) A second parameter passed to the extend method of a class defines the class interface.
213  * @returns {object} SubClass The class with its own members and the inherited ones from Biojs.
214  * 
215  * @example 
216  * Biojs.MyComponent = Biojs.extend(
217  * { // instance 
218  *    constructor: function(options) {
219  *       // constructor code here  
220  *    },
221  *  
222  *    opt: { target: "divId" },
223  *    
224  *    eventTypes: [ "myEvent1", "myEvent2" ],
225  *      
226  *    getVersion: function() {
227  *        return Biojs.MyComponent.VERSION;
228  *    }
229  *  }, 
230  *  { // class interface
231  *     VERSION: "3.14.15"
232  *  });
233  * 
234  *  alert(Biojs.MyComponent.VERSION);
235  * 
236  */
237 Biojs.extend = function(_child, _static) { // subclass
238 	var extend = Biojs.prototype.extend;
239 	
240 	// build the prototype
241 	Biojs._prototyping = true;
242 	
243 	/**
244 	 * @name proto
245 	 * @constructs
246 	 */
247 	var proto = new this;
248 
249 	// Inherit parent' events to the child
250 	if (proto.eventTypes instanceof Array) {
251 		for ( var i in proto.eventTypes ) {
252 			_child.eventTypes.push(proto.eventTypes[i]);
253 		}
254 	}
255 	
256 	// Inherit parent' options to the child
257 	if (proto.opt instanceof Object) {
258 		for ( var key in proto.opt ) {
259 			_child.opt[key] = proto.opt[key];
260 		}
261 	}
262 	
263 	extend.call(proto, _child);
264 	
265 	/**
266 	 * @ignore
267 	 */
268 	proto.base = function() {
269 		// call this method from any other method to invoke that method's ancestor
270 	};
271 
272 	delete Biojs._prototyping;
273 	
274 	// create the wrapper for the constructor function
275 	var constructor = proto.constructor;
276 	var klass = proto.constructor = function() {
277 		
278 		if (!Biojs._prototyping) {
279 
280 			if (this.constructor == klass) { // instantiation
281 
282 				// Create a instance of this class
283 				function BiojsComponent() {};
284 				BiojsComponent.prototype = proto;
285 				var instance = new BiojsComponent();
286 
287 				// Change the default option's values 
288 				// in the instance by the provided ones
289 				instance.setOptions(arguments[0]);
290 				
291 				// Set the event handlers
292 				instance.setEventHandlers(instance.eventTypes);
293 				
294 				// Set the unique id for the instance
295 				instance.biojsObjectId = Biojs.uniqueId();
296 				
297 				// register instance
298 				Biojs.addInstance(instance);
299 				
300 				// execute the instance's constructor
301 				constructor.apply(instance, arguments);
302 				
303 				// return the instance
304 				return instance;
305 				
306 			} else { // Calling to ancestor's constructor
307 				constructor.apply(this,arguments);
308 			}
309 		}
310 	};
311 
312 	// build the class interface
313 	klass.ancestor = this;
314 	klass.extend = this.extend;
315 	klass.forEach = this.forEach;
316 	klass.implement = this.implement;
317 	klass.prototype = proto;
318 	/**
319 	 * @ignore
320 	 */
321 	klass.valueOf = function(type) {
322 		return (type == "object") ? klass : constructor.valueOf();
323 	};
324 	klass.toString = this.toString;
325 	extend.call(klass, _static);
326 	
327 	// class initialization
328 	if (typeof klass.init == "function") {
329 		klass.init();
330 	}
331 	
332 	return klass;
333 };

335 Biojs.prototype = 
336 /** @lends Biojs# */
337 {
338 	extend: function(source, value) {	
339 		if (arguments.length > 1) { // extending with a name/value pair
340 			var ancestor = this[source];
341 			if (ancestor && (typeof value == "function") && // overriding a method?
342 				// the valueOf() comparison is to avoid circular references
343 				(!ancestor.valueOf || ancestor.valueOf() != value.valueOf()) &&
344 				/\bbase\b/.test(value)) {
345 				// get the underlying method
346 				var method = value.valueOf();
347 				// override
348 				value = function() {
349 					var previous = this.base || Biojs.prototype.base;
350 					this.base = ancestor;
351 					var returnValue = method.apply(this, arguments);
352 					this.base = previous;
353 					return returnValue;
354 				};
355 				// point to the underlying method
356 				value.valueOf = function(type) {
357 					return (type == "object") ? value : method;
358 				};
359 				value.toString = Biojs.toString;
360 			}
361 			this[source] = value;
362 		} else if (source) { // extending with an object literal
363 			var extend = Biojs.prototype.extend;
364 			// if this object has a customised extend method then use it
365 			if (!Biojs._prototyping && typeof this != "function") {
366 				extend = this.extend || extend;
367 			}
368 			var proto = {toSource: null};
369 			// do the "toString" and other methods manually
370 			var hidden = ["constructor", "toString", "valueOf"];
371 			// if we are prototyping then include the constructor
372 			var i = Biojs._prototyping ? 0 : 1;
373 			while (key = hidden[i++]) {
374 				if (source[key] != proto[key]) {
375 					extend.call(this, key, source[key]);
376 				}
377 			}
378 			// copy each of the source object's properties to this object
379 			for (var key in source) {
380 				if (!proto[key]) extend.call(this, key, source[key]);
381 			}	
382 		}
383 		
384 		return this;
385 	},
386 	
387 	/**
388 	 * Register a function under an event type in order to execute it whenever the event is triggered.
389 	 * @param {string} eventType The event to be listened.
390 	 * @param {function} actionPerformed The action to be executed whenever the event occurs. it 
391 	 * 
392 	 * 
393 	 * @example 
394 	 * 
395 	 * var listener = function(eventObj){
396 	 *    alert("Selected: "+eventObj.start+", end: "+ eventObj.end);
397 	 * }
398 	 * 
399 	 * var mySequence = new Biojs.Sequence( {
400 	 *    sequence : "mlpglallllaawtaralevptdgnagllaepqiamfcgrlnmhmnvqngsgtktcidtkegilqy",
401 	 *    target : "#div0001",
402 	 *    format : 'CODATA',
403 	 *    id : 'P918283'	
404 	 * });
405 	 * 
406 	 * mySequence.addListener('onSelectionChanged', listener);
407 	 * 
408 	 * // HTML div tag with the id='div0001' must exist in the HTML document 
409 	 * 
410 	 */
411 	addListener: function(eventType, actionPerformed) {
412 		if (this._eventHandlers) {
413 			// register the listener in this._eventHandlers for the eventType  
414 			for(var key in this._eventHandlers) {
415 				if ( eventType == this._eventHandlers[key].eventType ) {
416 					this._eventHandlers[key].addListener( actionPerformed );
417 					return;
418 				}
419 			} 
420 		}
421 	},
422 	
423 	/**
424 	 * Sets an event handler and an alias method for each string in the array eventTypes.
425 	 * This method is executed automatically before constructing an instance, using the eventTypes array 
426 	 * that should be defined as member of subclass. Then, the resulting instance will have methods 
427 	 * named in the form instance.≶eventName>(actionPerformed) for all eventTypes.
428 	 * 
429 	 * @param {string[]} eventTypes Array of names of the events to be set.
430 	 * 
431 	 */
432 	setEventHandlers: function (eventTypes) {
433 		// Supposed that this._eventHandlers does not exist.
434 		this._eventHandlers = [];
435 		// Because the event handlers are not initialized yet
436 		
437 		var alias = function (handler) {
438 			return function (actionPerformed) {
439 				handler.listeners.push(actionPerformed);
440 			} 
441 		};
442 		
443 		if ( typeof eventTypes == "object" ) {
444 			// Create an event handler for each eventType in eventTypes
445 			for ( var i=0; i < eventTypes.length; i++ ) {
446 				var handler = new Biojs.EventHandler( eventTypes[i] );
447 				this._eventHandlers.push( handler );
448 				// Creates the alias this.<eventType> (<actionPerformed>)
449 				// as alternative to be used instead of this.addistener(<eventType>, <actionPerformed>)
450 				
451 				this[ eventTypes[i] ] = new alias(handler);
452 			}
453 		} 
454 	}, 
455 	
456 	/**
457 	 * Trigger the registered functions under an event type.
458 	 * @param {string} eventType The event to be raised.
459 	 * @param {Object} params The values to be included into Biojs.Event object. 
460 	 * 
461 	 * @example 
462 	 * 
463 	 * Biojs.MyComponent = Biojs.extend({
464 	 *    // ...
465 	 *    // code before the event 
466 	 *    
467 	 *    this.raiseEvent('onSelectionChanged', {start : start, end : end});
468 	 *    
469 	 *    // code after the event 
470 	 *    // ... 
471 	 * });
472 	 * 
473 	 */
474 	raiseEvent : function(eventType, eventObj) {
475 		for(var key in this._eventHandlers ) {
476 			if ( eventType == this._eventHandlers[key].eventType ) {
477 				this._eventHandlers[key].triggerEvent( eventObj );
478 				return;
479 			}
480 		}
481 	},
482 	
483 	
484 	//
485 	// Save the option values to be applied to this component
486 	// 	options -> [Object] containing the values
487 	//	
488 	setOptions : function (options) {
489 		if ( this.opt instanceof Object )
490 		{
491 			this.opt = Biojs.Utils.clone(this.opt);
492 			for ( var key in options ) {
493 				this.opt[key] = options[key];
494 			}
495 		} 
496 	},
497 	
498 	// 
499 	// 
500 	// 	source -> [BioJs] the another component 
501 	// 	eventType -> [string] the event to be listened 
502 	// 	callbackFunction -> [function] the action to be executed
503 	//
504 	/**
505 	 * Connect this component with another by means listening its events.
506 	 * @param {Biojs} source The another component.
507 	 * @param {string} eventType The event to be listened.	 
508 	 * @param {function} actionPerformed The action to be executed whenever the event occurs. it 
509 	 * 
510 	 * 
511 	 * @example 
512 	 * 
513 	 * var mySequence = new Biojs.Sequence( {
514 	 *    sequence : "mlpglallllaawtaralevptdgnagllaepqiamfcgrlnmhmnvqngsgtktcidtkegilqy",
515 	 *    target : "#div0001",
516 	 *    format : 'CODATA',
517 	 *    id : 'P918283'	
518 	 * });
519 	 * 
520 	 * var anotherSequence = new Biojs.Sequence({
521 	 *    sequence : "laawtaralevptmlpglallldgnagllaepqi",
522 	 *    target : "#div0002",	
523 	 * });
524 	 * 
525 	 * anotherSequence.listen(
526 	 *    mySequence,
527 	 *    "onSelectionChange",
528 	 *    function( eventObj ) {
529 	 *       anotherSequence.setSelection(eventObj.start, eventObj.end);
530 	 *    }   
531 	 * );
532 	 * 
533 	 */
534 	listen: function ( source, eventType, callbackFunction ) {		
535 		if ( source instanceof Biojs ){
536 			if ( typeof callbackFunction == "function" ) {
537 				source.addListener(eventType, callbackFunction);
538 			} 
539 		} 
540 	},
541 	
542 	getId: function () {
543 		return this.biojsObjectId;
544 	}
545 	
546 };
