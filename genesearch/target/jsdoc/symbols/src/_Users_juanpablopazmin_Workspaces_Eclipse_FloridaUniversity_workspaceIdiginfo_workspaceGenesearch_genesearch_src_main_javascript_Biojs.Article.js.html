 91 	setData: function ( dataRow ) {
 92 		var self = this;
 93 		
 94 		var seeAlso = dataRow[2].split(';');
 95 		var interval = ( dataRow[9].length > 0 ) ? ', ' + dataRow[9] + '-' + dataRow[10] : '';
 96 		var sameAs = dataRow[3].split(';');
 97 		
 98 		this._title.html( dataRow[1] );
 99 		this._authors.html( dataRow[11] );
100 		this._summary.html( '<strong>' + dataRow[6] + '</strong>, '+ dataRow[4] +', Volume ' + dataRow[7] + ' (' + dataRow[8] + ')'+ interval );
101 		this._abstract.html( '<strong>Abstract: </strong>' +dataRow[5] );	
102 		this._seeAlso.html('');
103 		this._sameAs.html('');
104 		this._annotations.html('Loading...').css('text-align','left');
105 		this._references.html('Loading...');
106 
107 		jQuery.each( seeAlso, function ( i, value ) {
108 			self._seeAlso.append('<li><a href="' + value + '">' + value + '</a></li>');
109 		});
110 
111 		jQuery.each( sameAs, function ( i, value ) {
112 			var prefix = ( value.indexOf('doi') != -1 )? "DOI:" : ( value.indexOf('pubmed') != -1 )? "pmid:" : "";
113 			self._sameAs.append('<li><a href="' + value + '">' + prefix + value.substr( value.lastIndexOf("/") + 1 ) + '</a></li>');
114 		});
115 		
116 		
117 		this._setAnnotations( dataRow[0], this._annotations ); 
118 		this._setReferences( dataRow[0], this._references );

123 	_setAnnotations: function ( resourceURI, node ) {
124 		
125 		var terms;
126 		
127 		this._requestAnnotations( resourceURI, function (data) {
128 			
129 			this._annotations.html('').css('text-align','center');
130 			
131 			terms = this._decodeAnnotations(data);
132 			
133 			jQuery.each( terms, function ( term, topics ) {
134 				var termNode = jQuery('<li class="term"><span class="name">' + term.toLowerCase() + '</span></li>');
135 				var topicList = jQuery('<span class="topics" style="display:none"></span>').appendTo( termNode ); 
136 				var pattern=/(#|_|:|\/)([A-Za-z0-9])*$/g;
137 				var topic, topicKey, identifier, freq = 0;
138 				
139 				for ( t in topics ) {
140 					
141 					try {
142 
143 						topicKey = topics[t].replace(pattern, '');
144 						topic = Biojs.Article.TOPIC[ topicKey ];
145 						
146 						identifier = ( topics[t].match(pattern)[0] ).substr(1);
147 						
148 						jQuery('<a class="topic ' + topic.shortName + '">' + topic.prefix + identifier + '</a>')
149 							.attr({
150 								href: topic.baseUrl + identifier,
151 								topicType: topic.shortName,
152 								topicId: identifier
153 							})
154 							.appendTo(topicList);
155 						
156 						freq++;
157 						
158 					} catch ( e ) {
159 						Biojs.console.log("Error searching topic for " + topics[t] + ". message:" + e.message);
160 					}
161 
162 				}
163 				
164 				if ( topicList.children().length > 0 ) {
165 					topicList.last().addClass("last");
166 					termNode.appendTo(node);
167 				}
168 				
169 				termNode.children(':first-child')
170 					.css({
171 						"fontSize": (freq / 10 < 1) ? freq / 10 + 1 + "em": (freq / 10 > 2) ? "2em" : freq / 10 + "em",
172 						"cursor": 'pointer'
173 					})
174 					.click( function() { 
175 						node.find('.topics').hide();
176 						topicList.toggle(); 
177 					})
178 					.append('<span style="vertical-align:super; font-size:10px;">' + freq + '</super>');  
179 
180 			});
181 			
182 			Biojs.console.log("Raising event "+Biojs.Article.EVT_ON_ANNOTATIONS_LOADED);
183 			
184 			this.raiseEvent(Biojs.Article.EVT_ON_ANNOTATIONS_LOADED, { uri: resourceURI });
185 			
186 		});
187 			
188 	},

190 	_requestAnnotations: function ( resourceURI, actionCb ) {
191 
192 		strQuery = 
193 			'select ?annot fn:upper-case(str(?body)) AS ?term ?topic str(?init) AS ?posInit str(?end) AS ?posEnd ?comment '+
194 			'   where { '+
195 			'     ?annot a aot:ExactQualifier ; '+ //no mostrar, m�ltiple, sirve para agrupamiento de primer nivel, se puede omitir
196 			'     ao:annotatesResource <' + resourceURI + '>;'+
197 			'     ao:body ?body ; '+ //m�ltiple, se usa para el agrupamiento de primer nivel
198 			'     ao:hasTopic ?topic ;'+  //m�ltiple, se usa parcialmente para agrupamiento de segundo nivel, la combinaci�n body+topic es �nica
199 			'     ao:context ?context . '+ //no mostrart, no est� incluido en los resultados
200 			'     ?context rdfs:resource ?resource . '+ //no mostar, no est� incluido en los resultados
201 			'     OPTIONAL {?context aos:init ?init }. '+ //�nico, posici�n inicial de la anotaci�n
202 			'     OPTIONAL {?context aos:end ?end } . '+ //�nico, posici�n final de la anotaci�n 
203 			'     ?resource rdfs:comment ?comment .  '+ //�nico, contiene el texto donde aparece la anotaci�n
204 			'     FILTER (regex(str(?topic), "(uniprot|GO|CHEBI|ICD9|UMLS|fma|MSH|PO|MDDB|NDDF|NDFRT|NCBITaxon)")) . '+
205 			'} ORDER BY ?term ?topic';
206 
207 		this._doSparqlQuery( this.opt, strQuery, 'application/json', actionCb );
208 	},

210 	_setReferences: function ( resourceURI, node ) {
211 		
212 		var references;
213 		
214 		this._requestReferences( resourceURI, function (data) {
215 			
216 			references = this._decodeReferences(data);
217 			
218 			if ( Biojs.Utils.isEmpty( references ) ) {
219 				jQuery( node ).html( 'No references available' );
220 				
221 			} else {
222 				jQuery.each( references, function ( refId, ref ) {
223 					
224 					var interval = "";
225 					
226 					if ( ref.complete ) {
227 						interval = ( ref.start.length > 0 ) ? ', ' + ref.start + '-' + ref.end : '';
228 						'<li>' + ref.author + '. <strong>' + ref.jornalTitle + '</strong>, '+ ref.data +', Volume ' + ref.volume + ' (' + ref.issue + ')'+ interval +'</li>';
229 
230 					} else {
231 						jQuery( node ).append( '<li>' + ref.title + '</li>' );
232 					}
233 
234 				});
235 			}
236 			
237 		});
238 	},

240 	_requestReferences: function ( resourceURI, actionCb ) {
241 
242 		strQuery = 
243 			'select fn:substring(?ref, 82) AS ?refId ?title ?sameAs ?date ?journalTitle ?volume ?issue ?start ?end fn:concat(?authorLastName, ", ", ?authorFirstName) AS ?author ' +
244 			'where { ' +
245 			'   { <' + resourceURI + '> bibo:cites ?ref.' +
246 			'      ?ref dcterms:title ?title ;  ' +              //unique
247 			'      owl:sameAs ?sameAs ;  ' +                     //multiple
248 			'      dcterms:publisher ?publisher ;  ' +           //unique but not to show              
249 			'      bibo:volume ?volume ;  '  +                   //unique
250 			'      bibo:pageStart ?start ; ' +                   //unique
251 			'      bibo:pageEnd ?end ;  ' +                      //unique
252 			'      dcterms:issued ?date ;   ' +                  //unique
253 			'      bibo:authorList ?authorList . ' +             //unique but not to show
254 			'      OPTIONAL {?ref bibo:issue ?issue } . ' +      //unique, optional
255 			'      ?publisher dcterms:title ?journalTitle . ' +
256 			'      ?authorList rdfs:member ?member .  ' +        //multiple, not to show
257 			'      ?member foaf:givenName ?authorFirstName ; ' + //multiple
258 			'      foaf:familyName ?authorLastName. ' +          //multiple
259 			'   } UNION { <' + resourceURI + '> bibo:cites ?ref. ' +
260 			'      ?ref dcterms:title ?title . ' +
261 			'      OPTIONAL {?ref dcterms:publisher ?publisher } ' +
262 			'      FILTER (!bound(?publisher)) ' +
263 			'   } ' +
264 			'} ORDER BY ?refId ';
265 
266 		Biojs.console.log("SPARQL query for refs: " + strQuery );
267 		
268 		this._doSparqlQuery( this.opt, strQuery, 'application/json', actionCb );
269 	},

271 	_doSparqlQuery: function ( opt, query, format, actionCb ) {
272 		
273 		var self = this;
274 		var dataParams = [];
275 		
276 		dataParams.push({ name: "url", value: opt.sparqlUrl });
277 		dataParams.push({ name: "query", value: escape(query) }); 
278 		dataParams.push({ name: "format", value: format });
279 		
280 		Biojs.console.log("Requesting Query: " + query );
281 		
282 		jQuery.ajax({ 
283 			url: opt.proxyUrl,
284 			dataType: "text",
285 			data: dataParams,
286 			success: function ( data ) {
287 				Biojs.console.log("Data received");
288 				actionCb.call( self, data );
289 			}
290 		});
291 
292 	},

295 	_decodeAnnotations: function ( dataReceived ) {
296 		
297 		var decodedData = {};
298 		var results = [];
299 		var obj;
300 		var terms = {};
301 		
302 		try {
303 			decodedData = jQuery.parseJSON( dataReceived );
304 			results = decodedData.results.bindings;
305 			//Biojs.console.log( results );
306 
307 		} catch ( e ) {
308 			Biojs.console.log("Error decoding response data: " + e.message );
309 			return jsonData;
310 		}
311 		
312 		for ( r in results ) {
313 			obj = results[r];
314 			
315 			// Group by term. The terms objects should be like this:
316 			// terms {
317 			//   term1 : {topic1,topic2,...}
318 			//   term2 : {topic1,topic2,...}
319 			//   ...
320 			// }
321 			//
322 			if ( ! terms.hasOwnProperty( obj.term.value ) ) {
323 				terms[ obj.term.value ] = {};
324 			} 
325 			
326 			terms[ obj.term.value ][ obj.topic.value ] = obj.topic.value;
327 			
328 		}
329 		
330 		return terms;
331 		
332 	},

334 	_decodeReferences: function ( dataReceived ) {
335 		
336 		var decodedData = {};
337 		var results = [];
338 		var obj;
339 		var refs = {};
340 		
341 		try {
342 			
343 			decodedData = jQuery.parseJSON( dataReceived );
344 			results = decodedData.results.bindings;
345 			
346 			Biojs.console.log( "References:" );
347 			//Biojs.console.log( results );
348 			for ( r in results ) {
349 				obj = results[r];
350 				
351 				// Group by term. The refs objects should be like this:
352 				// refs {
353 				//   ref1 : { title: "..", sameAs: ".." ... }
354 				//   ref2 : { title: "..", sameAs: ".." ... }
355 				// }
356 				//
357 				if ( ! refs.hasOwnProperty( obj.refId.value ) ) {
358 					
359 					// Create a member object called refId
360 					var reference = refs[ obj.refId.value ] = {};
361 					reference.title = obj.title.value;
362 					
363 					try {
364 						// These values might either be present or not 
365 						reference.author = obj.author.value;
366 						reference.sameAs = obj.sameAs.value;
367 						reference.date = obj.date.value;
368 						reference.journalTitle = obj.journalTitle.value;
369 						reference.volume = obj.volume.value;
370 						reference.issue = obj.issue.value;
371 						reference.start = obj.start.value;
372 						reference.end = obj.end.value;
373 						reference.complete = true;
374 						
375 					} catch (e) { 
376 						reference.complete = false; continue; 
377 					}		
378 					
379 				} else {
380 					// object already created, copy multiple values sameAs and author
381 					reference.sameAs += ", " + obj.sameAs.value;
382 					reference.author += ", " + obj.author.value;
383 				}
384 
385 			}
386 
387 		} catch ( e ) {
388 			Biojs.console.log("Error decoding response data: " + e.message );
389 		}
390 		
391 		
392 		
393 		return refs;
394 		
395 	}
