To get the above Biojs.Table to work on your page, you need to do the following:

                    
                    
    
                        
  1. Include the file Biojs.Table.js and its dependencies to your page header.
                            
                            
                                
                                    
                                
                                    
                                
                                    
                                
                                    
                                
                                    
                                
                                    
                                
                              
                            
                            
                            
                                
    Biojs scripts
    
                                
    <script language="JavaScript" type="text/javascript" src="src/Biojs.js">

    
                                
                            
                            
                            
                                
    External script(s)
    
                                
    <script language="JavaScript" type="text/javascript" src="../biojs/dependencies/jquery/jquery-1.6.4.js">




    
                                
                            

                            
                                
    CSS stylesheet(s)
    
                                
    <link href="../biojs/css/biojs.Table.css" rel="stylesheet" type="text/css" />


    
                                
                            
                            
                            
                        
    2.     
                        
                        
  2. Create a div tag which holds an unique identifier.
                            
                                    <body>
                                ...
                                <div id="YourOwnDivId" />
                                ...
                                </body>
                            
    
                        
    3. 
                        
                        
  3. Create a code snippet within a <script> tag and instance Biojs.Table into.
                            
                                
                                    
                                    
    window.onload = function() {

   // Adds a flag to the cell value.
   var flagRender = function (col, dataRow, value) {
		  return ' ' + value;

		  return ( new Number(value) * 100 ).toFixed(2) + "%";

   // Example of instantiation of the Table
   var myTable = new Biojs.Table({
      target: "YourOwnDivId",
      hideColumns: [1],

      columns: [
