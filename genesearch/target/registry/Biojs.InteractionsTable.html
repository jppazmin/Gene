To get the above Biojs.InteractionsTable to work on your page, you need to do the following:

                    
                    
    
                        
  1. Include the file Biojs.InteractionsTable.js and its dependencies to your page header.
                            
                            
                                
                                    
                                
                                    
                                
                                    
                                
                                    
                                
                                    
                                
                                    
                                
                                    
                                
                              
                            
                            
                            
                                
    Biojs scripts
    
                                
    <script language="JavaScript" type="text/javascript" src="src/Biojs.js">


    
                                
                            
                            
                            
                                
    External script(s)
    
                                
    <script language="JavaScript" type="text/javascript" src="../biojs/dependencies/jquery/jquery-1.6.4.js">




    
                                
                            

                            
                                
    CSS stylesheet(s)
    
                                
    <link href="../biojs/css/biojs.Table.css" rel="stylesheet" type="text/css" />


    
                                
                            
                            
                            
                        
    2.     
                        
                        
  2. Create a div tag which holds an unique identifier.
                            
                                    <body>
                                ...
                                <div id="YourOwnDivId" />
                                ...
                                </body>
                            
    
                        
    3. 
                        
                        
  3. Create a code snippet within a <script> tag and instance Biojs.InteractionsTable into.
                            
                                
                                    
                                    
    window.onload = function() {

var myTable = new Biojs.InteractionsTable({
 	target: "YourOwnDivId",
 	dataSet: {
 		psicquicUrl: 'http://www.ebi.ac.uk/Tools/webservices/psicquic/intact/webservices/current/search/query',
 		proxyUrl: '../biojs/dependencies/proxy/proxy.php',
 		query: "brca2",
         version: "MITAB_VERSION_2_5_EXT",
         filter: false
     },
     rowSelection: true
});

};
